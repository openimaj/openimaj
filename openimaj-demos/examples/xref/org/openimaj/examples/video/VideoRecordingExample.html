30  package org.openimaj.examples.video;
31  
32  import java.awt.event.KeyAdapter;
33  import java.awt.event.KeyEvent;
34  import java.io.IOException;
35  
36  import javax.swing.SwingUtilities;
37  
38  import org.openimaj.image.MBFImage;
39  import org.openimaj.video.Video;
40  import org.openimaj.video.VideoDisplay;
41  import org.openimaj.video.VideoDisplayListener;
42  import org.openimaj.video.capture.VideoCapture;
43  import org.openimaj.video.xuggle.XuggleVideoWriter;
44  
45  /**
46   * Record the webcam to a file.
47   * 
48   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
49   */
50  public class VideoRecordingExample extends KeyAdapter implements VideoDisplayListener<MBFImage> {
51  	private Video<MBFImage> video;
52  	private VideoDisplay<MBFImage> display;
53  	private XuggleVideoWriter writer;
54  	private boolean close = false;
55  
56  	/**
57  	 * Default constructor
58  	 * 
59  	 * @throws IOException
60  	 */
61  	public VideoRecordingExample() throws IOException {
62  		// open webcam
63  		video = new VideoCapture(320, 240);
64  
65  		// open display
66  		display = VideoDisplay.createVideoDisplay(video);
67  
68  		// open a writer
69  		writer = new XuggleVideoWriter("video.flv", video.getWidth(), video.getHeight(), 30);
70  
71  		// set this class to listen to video display events
72  		display.addVideoListener(this);
73  
74  		// set this class to listen to keyboard events
75  		SwingUtilities.getRoot(display.getScreen()).addKeyListener(this);
76  	}
77  
78  	@Override
79  	public void afterUpdate(VideoDisplay<MBFImage> display) {
80  		// Do nothing
81  	}
82  
83  	@Override
84  	public void beforeUpdate(MBFImage frame) {
85  		// write a frame
86  		if (!close) {
87  			writer.addFrame(frame);
88  		}
89  	}
90  
91  	@Override
92  	public void keyPressed(KeyEvent key) {
93  		// wait for the escape key to be pressed
94  		close = key.getKeyCode() == KeyEvent.VK_ESCAPE;
95  	}
96  
97  	/**
98  	 * Main method
99  	 * 
100 	 * @param args
101 	 *            ignored
102 	 * @throws IOException
103 	 */
104 	public static void main(String[] args) throws IOException {
105 		new VideoRecordingExample();
106 	}
107 }
