50  public class Features {
51  	public static void main(String[] args) throws IOException {
52  		final AvgBrightness brightness = new AvgBrightness();
53  		final Colorfulness colorfulness = new Colorfulness();
54  		final Naturalness naturalness = new Naturalness();
55  		final RMSContrast contrast = new RMSContrast();
56  		final RGBRMSContrast rgbcontrast = new RGBRMSContrast();
57  		final Sharpness sharpness = new Sharpness();
58  		final SharpnessVariation sharpnessVariation = new SharpnessVariation();
59  		final Saturation saturation = new Saturation();
60  		final SaturationVariation saturationVariation = new SaturationVariation();
61  
62  		final File[] files = new File("/Users/jon/Dropbox/features/images").listFiles(new FilenameFilter() {
63  			@Override
64  			public boolean accept(File dir, String name) {
65  				return name.endsWith(".jpg");
66  			}
67  		});
68  
69  		for (final File f : files) {
70  			final MBFImage image = ImageUtilities.readMBF(f);
71  			final FImage fimage = Transforms.calculateIntensity(image);
72  
73  			brightness.analyseImage(image);
74  			System.out.println(f.getName() + "\tbrightness\t" + brightness.getBrightness());
75  
76  			image.analyseWith(colorfulness);
77  			System.out.println(f.getName() + "\tcolorfulness\t" + colorfulness.getColorfulness());
78  
79  			naturalness.analyseImage(image);
80  			System.out.println(f.getName() + "\tnaturalness\t" + naturalness.getNaturalness());
81  
82  			contrast.analyseImage(fimage);
83  			System.out.println(f.getName() + "\trms-contrast\t" + contrast.getContrast());
84  
85  			rgbcontrast.analyseImage(image);
86  			System.out.println(f.getName() + "\trgb-rms-contrast\t" + rgbcontrast.getContrast());
87  
88  			sharpness.analyseImage(fimage);
89  			System.out.println(f.getName() + "\tsharpness\t" + sharpness.getSharpness());
90  
91  			sharpnessVariation.analyseImage(fimage);
92  			System.out.println(f.getName() + "\tsharpness-variation\t" + sharpnessVariation.getSharpnessVariation());
93  
94  			saturation.analyseImage(image);
95  			System.out.println(f.getName() + "\tsaturation\t" + saturation.getSaturation());
96  
97  			saturationVariation.analyseImage(image);
98  			System.out.println(f.getName() + "\tsaturation-variation\t" + saturationVariation.getSaturationVariation());
99  		}
100 	}
