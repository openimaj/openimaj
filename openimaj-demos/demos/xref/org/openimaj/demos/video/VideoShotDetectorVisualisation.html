53  /**
54   * Demonstration of the OpenIMAJ HistogramVideoShotDetector and visualisation thereof.
55   * 
56   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
57   * @created 1 Jun 2011
58   */
59  @Demo(
60  		author = "David Dupplaw",
61  		description = "Gives a demo of the video shot detector by displaying an " +
62  				"animated visualisation of the process of shot detection using " +
63  				"differential histograms.",
64  		keywords = { "video", "shots", "shot detector" },
65  		title = "Video Shot Detector")
66  public class VideoShotDetectorVisualisation {
67  	/**
68  	 * Testing code.
69  	 * 
70  	 * @param args
71  	 */
72  	public static void main(final String[] args) {
73  		DisplayUtilities.displayName(new MBFImage(100, 100, 3), "vsd", true);
74  		DisplayUtilities.createNamedWindow("video").setLocation(0, 250);
75  
76  		final List<VideoKeyframe<MBFImage>> keyframes =
77  				new ArrayList<VideoKeyframe<MBFImage>>();
78  		final int th = 64;
79  		final int tw = 64;
80  		final int h = 200;
81  		final int w = Toolkit.getDefaultToolkit().getScreenSize().width - tw;
82  		final MBFImage m = new MBFImage(w + tw, h, 3);
83  		final MBFImageRenderer renderer = m.createRenderer();
84  		final ResizeProcessor rp = new ResizeProcessor(tw, th, true);
85  
86  		final XuggleVideo video = new XuggleVideo(
87  				"src/main/resources/org/openimaj/demos/video/guy_goma.mp4" );
88  		final CombiShotDetector vsd = new CombiShotDetector( video );
89  		vsd.addVideoShotDetector( new HistogramVideoShotDetector( video ), 1 );
90  		vsd.addVideoShotDetector( new LocalHistogramVideoShotDetector( video, 20 ), 1 );
91  		final double threshold = vsd.getThreshold();
92  		vsd.setStoreAllDifferentials(true);
93  		vsd.setFindKeyframes(true);
94  		vsd.addShotDetectedListener(new ShotDetectedListener<MBFImage>()
95  		{
96  			private double lastMax = 10000;
97  
98  			@Override
99  			public void shotDetected(final ShotBoundary<MBFImage> sb, final VideoKeyframe<MBFImage> vk)
100 			{
101 				// Store the keyframe
102 				if (vk != null)
103 					keyframes.add(vk.clone());
104 
105 				// Reset the image
106 				m.zero();
107 
108 				// Calculate the various variables required to draw the
109 				// visualisation.
110 				final DoubleFV dfv = vsd.getDifferentials();
111 				double max = Double.MIN_VALUE;
112 				for (int x = 0; x < dfv.length(); x++)
113 					max = Math.max(max, dfv.get(x));
114 				if (max > 50)
115 					this.lastMax = max;
116 
117 				// Draw all the keyframes found onto the image
118 				for (final VideoKeyframe<MBFImage> kf : keyframes)
119 				{
120 					final long fn = kf.getTimecode().getFrameNumber();
121 					final int x = (int) (fn * w / dfv.length());
122 
123 					// We draw the keyframes along the top of the visualisation.
124 					// So we draw a line to the frame to match it up to the
125 					// differential
126 					renderer.drawLine(x, h, x, 0, new Float[] { 0.3f, 0.3f, 0.3f });
127 					renderer.drawImage(kf.getImage().process(rp), x + 1, 0);
128 				}
129 
130 				// This is the threshold line drawn onto the image.
131 				renderer.drawLine(0, (int) (h - h / max * threshold), w,
132 						(int) (h - h / max * threshold), RGBColour.RED);
133 
134 				// Now draw all the differentials
135 				int x = 0;
136 				for (int z = 0; z < dfv.length(); z++)
137 				{
138 					x = z * w / dfv.length();
139 					renderer.drawLine(x, h, x, (int) (h - h / max * dfv.get(z)),
140 							RGBColour.WHITE);
141 				}
142 
143 				// Display the visualisation
144 				// DisplayUtilities.updateNamed( "vsd", m, "Video Shot Detector"
145 				// );
146 
147 				// System.out.println( "Keyframes: "+keyframes );
148 				// DisplayUtilities.display( "Keyframes: ", keyframes.toArray(
149 				// new Image<?,?>[0] ) );
150 			}
151 
152 			@Override
153 			public void differentialCalculated(final VideoTimecode vt, final double d, final MBFImage frame)
154 			{
155 
156 				// Display the visualisation
157 				// DisplayUtilities.updateNamed( "vsd", m, "Video Shot Detector"
158 				// );
159 				this.shotDetected(null, null);
160 
161 				renderer.drawShapeFilled(new Rectangle(w + tw / 2 - 5, th, 10, h - th), RGBColour.BLACK);
162 				renderer.drawLine(w + tw / 2, h, w + tw / 2, (int) (h - h / this.lastMax * d), 10,
163 						RGBColour.RED);
164 				renderer.drawImage(frame.process(rp), w, 0);
165 
166 				// Display the visualisation
167 				DisplayUtilities.updateNamed("vsd", m, "Video Shot Detector");
168 
169 				DisplayUtilities.displayName(frame, "video");
170 			}
171 		});
172 
173 		vsd.process();
174 	}
