64  		try {

67  			throw new RuntimeException(e);

71  	@Override
72  	public void analyseImage(FImage image) {
73  		CLQueue queue = context.createDefaultQueue();
74  
75  		CLImage2D in = CLImageConversion.convert(context, image);
76  		
77  		CLImageFormat outFmt = new CLImageFormat(ChannelOrder.RGBA, ChannelDataType.Float);
78  		CLImage2D out = context.createImage2D(CLMem.Usage.Output, outFmt, in.getWidth(), in.getHeight());
79  		
80  		kernel.setArgs(in, out);
81  		CLEvent evt = kernel.enqueueNDRange(queue, new int[] {(int) in.getWidth(), (int) in.getHeight()});
82  
83  		MBFImage res = CLImageConversion.convert(queue, evt, out, new MBFImage(image.width, image.height, 4));
84  		
85  		this.magnitude = res.bands.get(0);
86  		this.orientation = res.bands.get(1);
87  		this.gradientX = res.bands.get(2);
88  		this.gradientY = res.bands.get(3);
89  		
90  		in.release();
91  		out.release();
92  		queue.release();
93  	}
