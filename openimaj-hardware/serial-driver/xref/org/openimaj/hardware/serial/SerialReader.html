18  /**
19   * An event listener that receives data from the serial port, buffers the data,
20   * parses the data then calls the listeners for every sentence parsed.
21   *
22   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
23   *
24   * @created 12 Jul 2011
25   */
26  public class SerialReader implements SerialPortEventListener, Closeable
27  {
28  	/** The input stream from the serial device */
29  	private InputStream inputStream = null;
30  
31  	/** The parser being used for incoming data */
32  	private SerialDataParser parser = null;
33  
34  	/** We use trove to buffer the incoming data */
35  	private TByteList buffer = new TByteArrayList();
36  
37  	/** The maximum size of a buffer before parsing data */
38  	private int maxSize = 256;
39  
40  	/** Listeners */
41  	private List<SerialDataListener> listeners = new ArrayList<SerialDataListener>();
42  
43  	private boolean closed = false;
44  
45  	/**
46  	 * Default constructor
47  	 *
48  	 * @param in
49  	 * @param parser
50  	 */
51  	public SerialReader(InputStream in, SerialDataParser parser)
52  	{
53  		this.inputStream = in;
54  		this.parser = parser;
55  	}
56  
57  	/**
58  	 * {@inheritDoc}
59  	 */
60  	@Override
61  	public void serialEvent(SerialPortEvent event)
62  	{
63  		if (closed)
64  			return;
65  
66  		if (event != null && !event.isRXCHAR())
67  			return;
68  
69  		try
70  		{
71  			// Reads all the data from the serial port event (upto a maximum
72  			// size)
73  			int data = 0;
74  			while (buffer.size() < maxSize && (data = inputStream.read()) > -1)
75  				buffer.add((byte) data);
76  
77  			// Parse the data
78  			final String dataString = new String(buffer.toArray(), 0, buffer.size());
79  			final String[] strings = parser.parse(dataString);
80  			final String leftOvers = parser.getLeftOverString();
81  
82  			// If we've got to the end of the stream, we'll simply fire the
83  			// events
84  			// for the strings that are parsed and the left overs.
85  			if (data == -1)
86  			{
87  				if (strings.length > 0)
88  					fireDataReceived(strings);
89  				if (leftOvers.length() > 0)
90  					fireDataReceived(new String[] { leftOvers });
91  				buffer.clear();
92  			}
93  			else
94  			{
95  				// Keep the left-over parts of the string in the buffer
96  				if (leftOvers != null)
97  					buffer = buffer.subList(
98  							buffer.size() - leftOvers.length(),
99  							buffer.size());
100 				else
101 					buffer.clear();
102 
103 				// Let everyone know we have data!
104 				fireDataReceived(strings);
105 			}
106 		} catch (final IOException e)
107 		{
108 			// FIXME: RuntimeException? Seems a bit harsh.
109 			throw new RuntimeException(e);
110 		}
111 	}

126 	 * Remove the given listener from this reader.
127 	 *
128 	 * @param listener
129 	 *            The listener

133 		listeners.remove(listener);

136 	/**
137 	 * Fire multiple events: one for each parsed string.
138 	 *
139 	 * @param strings
140 	 *            The strings parsed from the parser.
141 	 */
142 	protected void fireDataReceived(String[] strings)
143 	{
144 		for (final String s : strings)
145 			for (final SerialDataListener listener : listeners)
146 				listener.dataReceived(s);
147 	}

149 	/**

152 	 * can be up to 128 bytes, then the buffer should be at least 128 bytes. It
153 	 * can be larger.
154 	 *
155 	 * @param maxSize
156 	 *            The size of the buffer to use.
157 	 */
158 	public void setMaxBufferSize(int maxSize)
159 	{
160 		this.maxSize = maxSize;
161 	}
