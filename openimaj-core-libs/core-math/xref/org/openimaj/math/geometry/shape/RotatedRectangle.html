43  public class RotatedRectangle implements Shape, Cloneable {
44  	/**
45  	 * The width of the rotated rectangle (Note that this is different to the
46  	 * width returned by {@link #getWidth()} which is the width of the regular
47  	 * bounding box)

50  
51  	/**
52  	 * The height of the rotated rectangle (Note that this is different to the
53  	 * height returned by {@link #getHeight()} which is the height of the
54  	 * regular bounding box)
55  	 */
56  	public float height;
57  
58  	/**
59  	 * The rotation angle in radians
60  	 */
61  	public float rotation;
62  
63  	/**
64  	 * The x-ordinate of the centroid
65  	 */
66  	public float cx;
67  
68  	/**
69  	 * The y-ordinate of the centroid
70  	 */
71  	public float cy;
72  
73  	/**
74  	 * Construct with a regular {@link Rectangle} rotated about its centroid
75  	 * 
76  	 * @param regRect
77  	 *            the regular rectangle
78  	 * @param rotation
79  	 *            the rotation angle in radians
80  	 */
81  	public RotatedRectangle(Rectangle regRect, float rotation) {
82  		this.width = regRect.width;
83  		this.height = regRect.height;
84  		this.cx = regRect.x + regRect.width / 2f;
85  		this.cy = regRect.y + regRect.height / 2f;
86  		this.rotation = rotation;
87  	}

93  	 *            the x-ordinate of the centroid
94  	 * @param y

96  	 * @param width
97  	 *            the width
98  	 * @param height
99  	 *            the height
100 	 * @param rotation
101 	 *            the rotation
102 	 */
103 	public RotatedRectangle(double x, double y, double width, double height, double rotation) {
104 		this.cx = (float) x;

107 		this.height = (float) height;
108 		this.rotation = (float) rotation;

215 	@Override
216 	public Polygon asPolygon() {
217 		final float b = (float) Math.cos(rotation) * 0.5f;
218 		final float a = (float) Math.sin(rotation) * 0.5f;
219 
220 		final Point2dImpl[] pts = new Point2dImpl[4];
221 		pts[0] = new Point2dImpl();
222 		pts[0].x = cx - a * height - b * width;
223 		pts[0].y = cy + b * height - a * width;
224 
225 		pts[1] = new Point2dImpl();
226 		pts[1].x = cx + a * height - b * width;
227 		pts[1].y = cy - b * height - a * width;
228 
229 		pts[2] = new Point2dImpl();
230 		pts[2].x = 2 * cx - pts[0].x;
231 		pts[2].y = 2 * cy - pts[0].y;
232 
233 		pts[3] = new Point2dImpl();
234 		pts[3].x = 2 * cx - pts[1].x;
235 		pts[3].y = 2 * cy - pts[1].y;
236 
237 		return new Polygon(pts);
238 	}
