32  import org.openimaj.citation.annotation.Reference;
33  import org.openimaj.citation.annotation.ReferenceType;
34  import org.openimaj.math.matrix.algorithm.pca.SvdPrincipalComponentAnalysis;
35  import org.openimaj.math.statistics.normalisation.Normaliser;
36  
37  import Jama.Matrix;

39  /**
40   * The ZCA Whitening transform. Works like PCA whitening, but after variance
41   * normalisation, rotates the data back to the original orientation. The benefit
42   * is that the data will still "look-like" the original input to some extent.
43   *
44   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
45   */
46  @Reference(
47  		type = ReferenceType.Article,
48  		author = { "Anthony J. Bell", "Terrence J. Sejnowski" },
49  		title = "The `Independent Components' of Natural Scenes are Edge Filters.",
50  		year = "1997",
51  		journal = "VISION RESEARCH",
52  		pages = { "3327", "", "3338" },
53  		volume = "37")
54  public class ZCAWhitening extends PCAWhitening {
55  	/**
56  	 * Construct with the given variance regularization parameter and data
57  	 * normalisation strategy.
58  	 *
59  	 * @param eps
60  	 *            the variance normalisation regularizer (each principle
61  	 *            dimension is divided by sqrt(lamba + eps), where lamba is the
62  	 *            corresponding eigenvalue).
63  	 * @param ns
64  	 *            the normalisation to apply to each input data vector prior to
65  	 *            training the transform or applying the actual whitening.
66  	 */
67  	public ZCAWhitening(double eps, Normaliser ns) {
68  		super(eps, ns);
69  	}
70  
71  	@Override
72  	public void train(double[][] data) {
73  		ns.normalise(data);
74  		final double[][] normData = ns.normalise(data);
75  
76  		final SvdPrincipalComponentAnalysis pca = new SvdPrincipalComponentAnalysis();
77  		pca.learnBasisNorm(new Matrix(normData));
78  		transform = pca.getBasis().copy();
79  		final double[] weight = pca.getEigenValues();
80  		final double[][] td = transform.getArray();
81  
82  		for (int c = 0; c < weight.length; c++)
83  			weight[c] = 1 / Math.sqrt(weight[c] + eps);
84  
85  		for (int r = 0; r < td.length; r++)
86  			for (int c = 0; c < td[0].length; c++)
87  				td[r][c] = td[r][c] * weight[c];
88  
89  		transform = transform.times(pca.getBasis().transpose());
90  	}
91  }
