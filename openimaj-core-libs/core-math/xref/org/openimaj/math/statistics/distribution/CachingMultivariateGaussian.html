38  /**
39   * A single multidimensional Gaussian. This implementation computes the inverse
40   * and Cholesky decomposition of the covariance matrix and caches them for
41   * efficient sampling and probability computation.
42   * 
43   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
44   * 
45   */
46  public class CachingMultivariateGaussian extends AbstractMultivariateGaussian implements MultivariateGaussian {
47  	protected Matrix covar;
48  	protected int N;
49  
50  	private Matrix inv_covar;
51  	private double pdf_const_factor;
52  	private Matrix chol;
53  
54  	protected CachingMultivariateGaussian() {
55  	}
56  
57  	/**
58  	 * Construct the Gaussian with the provided center and covariance
59  	 * 
60  	 * @param mean
61  	 *            centre of the Gaussian
62  	 * @param covar
63  	 *            covariance of the Gaussian
64  	 */
65  	public CachingMultivariateGaussian(Matrix mean, Matrix covar) {
66  		N = mean.getColumnDimension();
67  		this.mean = mean;
68  		this.covar = covar;
69  		cacheValues();
70  	}
71  
72  	/**
73  	 * Construct the Gaussian with the zero mean and unit variance
74  	 * 
75  	 * @param ndims
76  	 *            number of dimensions
77  	 */
78  	public CachingMultivariateGaussian(int ndims) {
79  		N = ndims;
80  		this.mean = new Matrix(1, N);
81  		this.covar = Matrix.identity(N, N);
82  		cacheValues();
83  	}
84  
85  	protected void cacheValues() {
86  		inv_covar = covar.inverse();
87  		pdf_const_factor = 1.0 / Math.sqrt((Math.pow((2 * Math.PI), N) * covar.det()));
88  
89  		chol = covar.chol().getL();
90  	}
91  
92  	/**
93  	 * Estimate a multidimensional Gaussian from the data
94  	 * 
95  	 * @param samples
96  	 *            the data
97  	 * @return the Gaussian with the best fit to the data
98  	 */
99  	public static CachingMultivariateGaussian estimate(float[][] samples) {
100 		final int ndims = samples[0].length;
101 
102 		final CachingMultivariateGaussian gauss = new CachingMultivariateGaussian();
103 		gauss.N = ndims;
104 
105 		final MeanAndCovariance res = new MeanAndCovariance(samples);
106 		gauss.mean = res.mean;
107 		gauss.covar = res.covar;
108 
109 		gauss.cacheValues();
110 
111 		return gauss;
112 	}

121 	public static MultivariateGaussian estimate(Matrix samples) {

147 	/**
148 	 * Get the probability for a given point in space relative to the PDF
149 	 * represented by this Gaussian.
150 	 * 
151 	 * @param sample
152 	 *            the point
153 	 * @return the probability
154 	 */
155 	@Override
156 	public double estimateProbability(double[] sample) {
157 		final Matrix xm = new Matrix(1, N);
158 		for (int i = 0; i < N; i++)
159 			xm.set(0, i, sample[i] - mean.get(0, i));
160 
161 		final Matrix xmt = xm.transpose();
162 
163 		final double v = xm.times(inv_covar.times(xmt)).get(0, 0);
164 
165 		return pdf_const_factor * Math.exp(-0.5 * v);
166 	}

172 	 * @param sample

176 	public double estimateProbability(Float[] sample) {
177 		final Matrix xm = new Matrix(1, N);
178 		for (int i = 0; i < N; i++)
179 			xm.set(0, i, sample[i] - mean.get(0, i));
180 
181 		final Matrix xmt = xm.transpose();
182 
183 		final double v = xm.times(inv_covar.times(xmt)).get(0, 0);
184

188 	@Override
189 	public double estimateLogProbability(double[] sample) {
190 		final Matrix xm = new Matrix(1, N);
191 		for (int i = 0; i < N; i++)
192 			xm.set(0, i, sample[i] - mean.get(0, i));
193 
194 		final Matrix xmt = xm.transpose();
195 
196 		final double v = xm.times(inv_covar.times(xmt)).get(0, 0);
197 
198 		return Math.log(pdf_const_factor) + (-0.5 * v);
199 	}

224 	@Override
225 	public double[] sample(Random rng) {
226 		final Matrix vec = new Matrix(N, 1);
227 
228 		for (int i = 0; i < N; i++)
229 			vec.set(i, 0, rng.nextGaussian());
230 
231 		final Matrix result = this.mean.plus(chol.times(vec).transpose());
232 
233 		return result.getArray()[0];
234 	}

236 	@Override
237 	public double[][] sample(int count, Random rng) {
238 		final double[][] samples = new double[count][];
239 
240 		for (int i = 0; i < count; i++)
241 			samples[i] = sample(rng);
242 
243 		return samples;
244 	}
