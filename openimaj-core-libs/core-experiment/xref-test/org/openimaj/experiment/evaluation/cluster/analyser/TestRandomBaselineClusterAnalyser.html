67  		logger.debug("Loading groundtruth");
68  		this.groundTruth = new ClusterEvalDataset(TestRandomBaselineClusterAnalyser.class.getResourceAsStream(GROUNDTRUTH));
69  		logger.debug("Loading estimate");
70  		this.estimate = new ClusterEvalDataset(TestRandomBaselineClusterAnalyser.class.getResourceAsStream(ESTIMATES));
71  	}

73  	/**
74  	 * 
75  	 */
76  	@Test
77  	public void test(){
78  		logger.debug("Converting datasets to int array clusters");
79  		int[][] gt = this.groundTruth.toClusters();
80  		int[][] es = this.estimate.toClusters();
81  		
82  		logger.debug("Performing NMI random baseline analysis");
83  		RandomBaselineClusterAnalyser<NMIClusterAnalyser, NMIAnalysis> r = new RandomBaselineClusterAnalyser<NMIClusterAnalyser, NMIAnalysis>(new NMIClusterAnalyser());
84  		RandomBaselineClusterAnalysis<NMIAnalysis> ann = r.analyse(gt, es);
85  		logger.debug(ann.getSummaryReport());
86  		
87  		logger.debug("Performing purity random baseline analysis");
88  		RandomBaselineClusterAnalyser<PurityClusterAnalyser, PurityAnalysis> rpure = new RandomBaselineClusterAnalyser<PurityClusterAnalyser, PurityAnalysis>(new PurityClusterAnalyser());
89  		RandomBaselineClusterAnalysis<PurityAnalysis> annpure = rpure.analyse(gt, es);
90  		logger.debug(annpure.getSummaryReport());
91  		
92  		logger.debug("Performing randind random baseline analysis");
93  		RandomBaselineClusterAnalyser<RandomIndexClusterAnalyser, RandomIndexAnalysis> rrandi = new RandomBaselineClusterAnalyser<RandomIndexClusterAnalyser, RandomIndexAnalysis>(new RandomIndexClusterAnalyser());
94  		RandomBaselineClusterAnalysis<RandomIndexAnalysis> annrandi = rrandi.analyse(gt, es);
95  		logger.debug(annrandi.getSummaryReport());
96  	}
