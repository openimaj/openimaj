39  /**
40   * The normalised mutual information of a cluster estimate

42   * @author Sina Samangooei (ss@ecs.soton.ac.uk)

44  public class NMIClusterAnalyser implements ClusterAnalyser<NMIAnalysis> {

46  	private final static Logger logger = LogManager.getLogger(NMIClusterAnalyser.class);
47  
48  	@Override
49  	public NMIAnalysis analyse(int[][] correct, int[][] estimated) {
50  		final NMIAnalysis ret = new NMIAnalysis();
51  		final Map<Integer, Integer> invCor = ClusterAnalyserUtils.invert(correct);
52  		final Map<Integer, Integer> invEst = ClusterAnalyserUtils.invert(estimated);
53  		ret.nmi = nmi(correct, estimated, invCor, invEst);
54  		return ret;
55  	}
56  
57  	private double nmi(int[][] c, int[][] e, Map<Integer, Integer> ic, Map<Integer, Integer> ie) {
58  		final double N = Math.max(ic.size(), ie.size());
59  		final double mi = mutualInformation(N, c, e, ic, ie);
60  		LoggerUtils.debugFormat(logger, "Iec = %2.5f", mi);
61  		final double ent_e = entropy(e, N);
62  		LoggerUtils.debugFormat(logger, "He = %2.5f", ent_e);
63  		final double ent_c = entropy(c, N);
64  		LoggerUtils.debugFormat(logger, "Hc = %2.5f", ent_c);
65  		return mi / ((ent_e + ent_c) / 2);
66  	}
67  
68  	/**
69  	 * Maximum liklihood estimate of the entropy
70  	 * 
71  	 * @param clusters
72  	 * @param N
73  	 * @return
74  	 */
75  	private double entropy(int[][] clusters, double N) {
76  		double total = 0;
77  		for (int k = 0; k < clusters.length; k++) {
78  			LoggerUtils.debugFormat(logger, "%2.1f/%2.1f * log2 ((%2.1f / %2.1f) )", (double) clusters[k].length, N,
79  					(double) clusters[k].length, N);
80  			final double prop = clusters[k].length / N;
81  			total += prop * log2(prop);
82  		}
83  		return -total;
84  	}
85  
86  	private double log2(double prop) {
87  		if (prop == 0)
88  			return 0;
89  		return Math.log(prop) / Math.log(2);
90  	}

92  	/**
93  	 * Maximum Liklihood estimate of the mutual information
94  	 * 
95  	 * @param c
96  	 * @param e
97  	 * @param ic
98  	 * @param ie
99  	 * @return
100 	 */
101 	private double mutualInformation(double N, int[][] c, int[][] e, Map<Integer, Integer> ic, Map<Integer, Integer> ie) {
102 		double mi = 0;
103 		for (int k = 0; k < e.length; k++) {
104 			final double n_e = e[k].length;
105 			for (int j = 0; j < c.length; j++) {
106 				final double n_c = c[j].length;
107 				double both = 0;
108 				for (int i = 0; i < e[k].length; i++) {
109 					final Integer itemCluster = ic.get(e[k][i]);
110 					if (itemCluster == null)
111 						continue;
112 					if (itemCluster == j)
113 						both++;
114 				}
115 				final double normProp = (both * N) / (n_c * n_e);
116 				// LoggerUtils.debugFormat(logger,"normprop = %2.5f",normProp);
117 				final double sum = (both / N) * (log2(normProp));
118 				mi += sum;
119 
120 				// LoggerUtils.debugFormat(logger,"%2.1f/%2.1f * log2 ((%2.1f * %2.1f) / (%2.1f * %2.1f)) = %2.5f",both,N,both,N,n_c,n_e,sum);
121 			}
122 		}
123 		return mi;
124 	}
