33  package org.openimaj.experiment.evaluation.agreement;

35  import gnu.trove.map.hash.TObjectIntHashMap;
36  
37  import java.util.Map;
38  
39  /**
40   * Calculates the interrater agreement for a given dataset between two (and only
41   * two) raters.
42   * <p>
43   * Cohen's Kappa is defined as: (PrA - PrE) / (1 - PrE) where PrA is the
44   * percentage agreement, and PrE is the probability of random agreement. PrA =
45   * agreement / total and PrE = PrX + PrY where PrX and PrY are the probability
46   * of both agreeing on X or both agreeing on Y randomly (that is,
47   * Pr(r1,x)*Pr(r2,x) ... )
48   * 
49   * @see "http://en.wikipedia.org/wiki/Cohen's_kappa"
50   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
51   * @created 12 Aug 2013
52   */
53  public class CohensKappaInterraterAgreement
54  {
55  	/**
56  	 * The input should be a {@link Map} for each rater where the keys represent
57  	 * all the subjects that were rated by the raters and the values represent
58  	 * the annotations given by the raters. Agreement between the raters is
59  	 * determined by {@link #equals(Object)} for the INSTANCE type. Annotations
60  	 * for subjects which are not in both sets are ignored.
61  	 * 
62  	 * @see "http://en.wikipedia.org/wiki/Cohen's_kappa"
63  	 * 
64  	 * @param rater1
65  	 *            The annotations from rater 1
66  	 * @param rater2
67  	 *            The annotations from rater 2
68  	 * @return Cohen's Kappa [0,1]
69  	 */
70  	public static <K, A> double calculate(
71  			final Map<K, A> rater1,
72  			final Map<K, A> rater2)
73  	{
74  		int totalCount = 0;
75  		int agreementCount = 0;
76  		final TObjectIntHashMap<A> answerCountsR1 = new TObjectIntHashMap<A>();
77  		final TObjectIntHashMap<A> answerCountsR2 = new TObjectIntHashMap<A>();
78  
79  		for (final K subjectKey : rater1.keySet())
80  		{
81  			// We can only form an agreement if both raters rated this
82  			// specific subject, so let's check
83  			if (rater2.keySet().contains(subjectKey))
84  			{
85  				final A r1a = rater1.get(subjectKey);
86  				final A r2a = rater2.get(subjectKey);
87  
88  				// It's possible that the key exists but is mapped to
89  				// a null value (for example, if majority voting was used
90  				// to generate the set and there was no majority).
91  				if (r1a == null || r2a == null)
92  					continue;
93  
94  				// Get the answers from the raters
95  				final A annotation1 = r1a;
96  				final A annotation2 = r2a;
97  
98  				// Count the agreements
99  				if (annotation1.equals(annotation2))
100 					agreementCount++;
101 
102 				// Count each of the answers for each of the raters
103 				answerCountsR1.putIfAbsent(annotation1, 0);
104 				answerCountsR2.putIfAbsent(annotation2, 0);
105 				answerCountsR1.increment(annotation1);
106 				answerCountsR2.increment(annotation2);
107 
108 				// Keep a running total
109 				totalCount++;
110 			}
111 		}
112 
113 		System.out.println(answerCountsR1);
114 
115 		final double PrA = agreementCount / (double) totalCount;
116 		System.out.println(PrA);
117 
118 		double PrE = 0;
119 		for (final A ann : answerCountsR1.keySet())
120 		{
121 			final Integer i = answerCountsR2.get(ann);
122 			final double PrAnnR1 = answerCountsR1.get(ann) / (double) totalCount;
123 			final double PrAnnR2 = (i == null ? 0 : i) / (double) totalCount;
124 			PrE += PrAnnR1 * PrAnnR2;
125 		}
126 		System.out.println(PrE);
127 
128 		final double kappa = (PrA - PrE) / (1d - PrE);
129 
130 		return kappa;
131 	}
