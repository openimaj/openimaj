82  	/**
83  	 * Returns a <code>BufferedImage</code> as the result of decoding a supplied
84  	 * <code>File</code> with an <code>ImageReader</code> chosen automatically
85  	 * from among those currently registered. The <code>File</code> is wrapped
86  	 * in an <code>ImageInputStream</code>. If no registered
87  	 * <code>ImageReader</code> claims to be able to read the resulting stream,
88  	 * <code>null</code> is returned.
89  	 *
90  	 * <p>
91  	 * The current cache settings from <code>getUseCache</code>and
92  	 * <code>getCacheDirectory</code> will be used to control caching in the
93  	 * <code>ImageInputStream</code> that is created.
94  	 *
95  	 * <p>
96  	 * Note that there is no <code>read</code> method that takes a filename as a
97  	 * <code>String</code>; use this method instead after creating a
98  	 * <code>File</code> from the filename.
99  	 *
100 	 * <p>
101 	 * This method does not attempt to locate <code>ImageReader</code>s that can
102 	 * read directly from a <code>File</code>; that may be accomplished using
103 	 * <code>IIORegistry</code> and <code>ImageReaderSpi</code>.
104 	 *
105 	 * @param input
106 	 *            a <code>File</code> to read from.
107 	 *
108 	 * @return a <code>BufferedImage</code> containing the decoded contents of
109 	 *         the input, or <code>null</code>.
110 	 *
111 	 * @exception IllegalArgumentException
112 	 *                if <code>input</code> is <code>null</code>.
113 	 * @exception IOException
114 	 *                if an error occurs during reading.
115 	 */
116 	public static BufferedImage read(File input) throws IOException {
117 		if (input == null) {
118 			throw new IllegalArgumentException("input == null!");
119 		}
120 		if (!input.canRead()) {
121 			throw new IIOException("Can't read input file!");
122 		}
123 		InputStream stream = null;
124 		try {
125 			stream = new FileInputStream(input);
126 			return read(stream);
127 		} finally {
128 			try {
129 				stream.close();
130 			} catch (final IOException e) {
131 			}
132 		}
133 	}

136 	 * Returns a <code>BufferedImage</code> as the result of decoding a supplied
137 	 * <code>InputStream</code> with an <code>ImageReader</code> chosen
138 	 * automatically from among those currently registered. The
139 	 * <code>InputStream</code> is wrapped in an <code>ImageInputStream</code>.
140 	 * If no registered <code>ImageReader</code> claims to be able to read the
141 	 * resulting stream, <code>null</code> is returned.
142 	 *
143 	 * <p>
144 	 * The current cache settings from <code>getUseCache</code>and
145 	 * <code>getCacheDirectory</code> will be used to control caching in the
146 	 * <code>ImageInputStream</code> that is created.
147 	 *
148 	 * <p>
149 	 * This method does not attempt to locate <code>ImageReader</code>s that can
150 	 * read directly from an <code>InputStream</code>; that may be accomplished
151 	 * using <code>IIORegistry</code> and <code>ImageReaderSpi</code>.
152 	 *
153 	 * <p>
154 	 * This method <em>does not</em> close the provided <code>InputStream</code>
155 	 * after the read operation has completed; it is the responsibility of the
156 	 * caller to close the stream, if desired.
157 	 *
158 	 * @param input
159 	 *            an <code>InputStream</code> to read from.
160 	 *
161 	 * @return a <code>BufferedImage</code> containing the decoded contents of
162 	 *         the input, or <code>null</code>.
163 	 *
164 	 * @exception IllegalArgumentException
165 	 *                if <code>input</code> is <code>null</code>.
166 	 * @exception IOException
167 	 *                if an error occurs during reading.
168 	 */
169 	public static BufferedImage read(InputStream input) throws IOException {
170 		if (input == null) {
171 			throw new IllegalArgumentException("input == null!");
172 		}
173 
174 		final NonClosableInputStream buffer = new NonClosableInputStream(input);
175 		buffer.mark(100 * 1024 * 1024); // 100mb is big enough?
176 
177 		BufferedImage bi;
178 		try {
179 			bi = readInternal(buffer);
180 		} catch (final Exception ex) {
181 			bi = null;
182 		}
183 
184 		if (bi == null) {
185 			buffer.reset();
186 			try {
187 				bi = Sanselan.getBufferedImage(buffer);
188 			} catch (final Throwable e) {
189 				throw new IOException(e);
190 			}
191 		}
192 
193 		return bi;
194 	}

227 		if (input == null) {
228 			throw new IllegalArgumentException("input == null!");
229 		}
230

232 		try {

235 			throw new IIOException("Can't get input stream from URL!", e);
236 		}

238 		return read(istream);

241 	/**
242 	 * Returns a <code>BufferedImage</code> as the result of decoding a supplied
243 	 * <code>ImageInputStream</code> with an <code>ImageReader</code> chosen
244 	 * automatically from among those currently registered. If no registered
245 	 * <code>ImageReader</code> claims to be able to read the stream,
246 	 * <code>null</code> is returned.
247 	 *
248 	 * @param input
249 	 *            an <code>ImageInputStream</code> to read from.
250 	 *
251 	 * @return a <code>BufferedImage</code> containing the decoded contents of
252 	 *         the input, or <code>null</code>.
253 	 *
254 	 * @exception IllegalArgumentException
255 	 *                if <code>stream</code> is <code>null</code>.
256 	 * @exception IOException
257 	 *                if an error occurs during reading.
258 	 * @throws ImageReadException
259 	 */
260 	private static BufferedImage readInternal(BufferedInputStream binput) throws IOException, ImageReadException {
261 		if (binput == null) {
262 			throw new IllegalArgumentException("stream == null!");
263 		}
264 
265 		ImageInfo info;
266 		try {
267 			info = Sanselan.getImageInfo(binput, null);
268 		} catch (final ImageReadException ire) {
269 			info = null;
270 		} finally {
271 			binput.reset();
272 		}
273 
274 		if (info != null && info.getFormat() == ImageFormat.IMAGE_FORMAT_JPEG) {
275 			if (info.getColorType() == ImageInfo.COLOR_TYPE_CMYK) {
276 				final ImageReader reader = getMonkeyReader();
277 
278 				if (reader == null) {
279 					// fallback to the ImageIO reader... one day it might be
280 					// fixed
281 					return ImageIO.read(binput);
282 				} else {
283 					return loadWithReader(reader, binput);
284 				}
285 			} else {
286 				// first try JAI if it's available
287 				try {
288 					// OpenJDK7 doesn't work properly with JAI as some of the
289 					// classes are missing!! This next line will throw in such
290 					// cases:
291 					Class.forName("com.sun.image.codec.jpeg.ImageFormatException");
292 
293 					synchronized (JAI.class) {
294 						return JAI.create("stream", SeekableStream.wrapInputStream(binput, false)).getAsBufferedImage();
295 					}
296 				} catch (final Exception e) {
297 					// JAI didn't work... we'll fall back to ImageIO, but try
298 					// the monkey first
299 					binput.reset();
300 
301 					// First try the Monkey reader
302 					final ImageReader reader = getMonkeyReader();
303 
304 					if (reader == null) {
305 						// fallback to the ImageIO reader... one day it might be
306 						// fixed
307 						return ImageIO.read(binput);
308 					} else {
309 						try {
310 							return loadWithReader(reader, binput);
311 						} catch (final Exception ee) {
312 							// fallback to the ImageIO reader... one day it
313 							// might be
314 							// fixed
315 							return ImageIO.read(binput);
316 						}
317 					}
318 				}
319 			}
320 		} else {
321 			return ImageIO.read(binput);
322 		}
323 	}

325 	/**
326 	 * Load an image with the given reader
327 	 *
328 	 * @param reader
329 	 * @param binput
330 	 * @return
331 	 * @throws IOException
332 	 */
333 	private static BufferedImage loadWithReader(ImageReader reader, BufferedInputStream binput) throws IOException {
334 		final ImageInputStream stream = ImageIO.createImageInputStream(binput);
335 
336 		final ImageReadParam param = reader.getDefaultReadParam();
337 		reader.setInput(stream, true, true);
338 
339 		return reader.read(0, param);
340 	}

342 	/**
343 	 * Get the TwelveMonkeys reader if its present and attempt to load it if
344 	 * necessary.
345 	 *
346 	 * @return the TwelveMonkeys JPEG Reader or null if it can't be loaded.
347 	 */
348 	private static ImageReader getMonkeyReader() {
349 		Iterator<ImageReader> iter = ImageIO.getImageReadersByFormatName("jpeg");
350 		while (iter.hasNext()) {
351 			final ImageReader reader = iter.next();
352 			if (reader instanceof com.twelvemonkeys.imageio.plugins.jpeg.JPEGImageReader)
353 				return reader;
354 		}
355 
356 		IIORegistry.getDefaultInstance().registerServiceProvider(
357 				new com.twelvemonkeys.imageio.plugins.jpeg.JPEGImageReaderSpi());
358 
359 		iter = ImageIO.getImageReadersByFormatName("jpeg");
360 		while (iter.hasNext()) {
361 			final ImageReader reader = iter.next();
362 			if (reader instanceof com.twelvemonkeys.imageio.plugins.jpeg.JPEGImageReader)
363 				return reader;
364 		}
365 
366 		return null;
367 	}
