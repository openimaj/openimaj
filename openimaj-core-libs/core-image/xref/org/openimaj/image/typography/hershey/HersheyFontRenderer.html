82  	protected void drawText(final String text, final HersheyFontStyle<T> sty, final int xc, final int yc,
83  			final boolean Draw, final Rectangle r, final ImageRenderer<T, ?> renderer)
84  	{
85  		final HersheyFontData fnt = ((HersheyFont)sty.getFont()).data;
86  		int character;
87  		int len;
88  		int rotpx = 0, rotpy = 0;
89  		int xp, yp;
90  		boolean rotate = false;
91  		float cosTheta = 0, sinTheta = 0;
92  		float verticalOffsetFactor = 0;
93  
94  		// set the flag to true if the angle is not 0.0
95  		rotate = (sty.getAngle() != 0.0) ? true : false;
96  
97  		// if we are to do a rotation
98  		if (rotate) {
99  			// set up the rotation variables
100 			final float theta = -sty.getAngle();
101 			cosTheta = (float) Math.cos(theta);
102 			sinTheta = (float) Math.sin(theta);
103 
104 			// set the position to do all rotations about
105 			rotpx = xc;
106 			rotpy = yc;
107 		}
108 
109 		// starting position
110 		xp = xc;
111 		yp = yc;
112 
113 		switch (sty.getVerticalAlignment()) {
114 		case VERTICAL_TOP:
115 			verticalOffsetFactor = 0;
116 			break;
117 		case VERTICAL_HALF:
118 			verticalOffsetFactor = 0.5f;
119 			break;
120 		case VERTICAL_BOTTOM:
121 			verticalOffsetFactor = 1;
122 			break;
123 		case VERTICAL_CAP:
124 			verticalOffsetFactor = 0.25f;
125 			break;
126 		}
127 
128 		// move the y position based on the vertical alignment
129 		yp -= (int) (verticalOffsetFactor * (sty.getActualHeightScale() * (fnt.characterSetMaxY - fnt.characterSetMinY)));
130 
131 		// if we have a non-standard horizontal alignment
132 		if ((sty.getHorizontalAlignment() != HorizontalAlignment.HORIZONTAL_LEFT)) {
133 			// find the length of the string in pixels ...
134 			len = 0;
135 
136 			for (int j = 0; j < text.length(); j++) {
137 				// the character's number in the array ...
138 				character = text.charAt(j) - ' ';
139 
140 				len += (fnt.characterMaxX[character] - fnt.characterMinX[character]) * sty.getActualWidthScale();
141 			}
142 
143 			// if we are center aligned
144 			if (sty.getHorizontalAlignment() == HorizontalAlignment.HORIZONTAL_CENTER) {
145 				// move the starting point half to the left
146 				xp -= len / 2;
147 			} else {
148 				// alignment is right, move the start all the way to the left
149 				xp -= len;
150 			}
151 		}
152 
153 		// loop through each character in the string ...
154 		r.x = r.y = Integer.MAX_VALUE;
155 		for (int j = 0; j < text.length(); j++) {
156 			// the character's number in the array ...
157 			character = text.charAt(j) - ' ';
158 
159 			if (character < 0)
160 				character = ' ';
161 
162 			// render this character
163 			this.drawCharacter(xp, yp, rotpx, rotpy, sty.getActualWidthScale(), sty.getActualHeightScale(), rotate,
164 					sinTheta, cosTheta, Draw, fnt.characterVectors[character],
165 					fnt.numberOfPoints[character], fnt.characterMinX[character],
166 					fnt.characterSetMinY, sty.getStrokeWidth(), sty.isItalic(), sty.getItalicSlant(), renderer,
167 					sty.getColour(), r);
168 
169 			// advance the starting coordinate
170 			final int actualWidth = (int) ((fnt.characterMaxX[character] - fnt.characterMinX[character]) * sty
171 					.getActualWidthScale());
172 			xp += actualWidth;
173 
174 			// r.width += actualWidth;
175 		} // end for each character
176 
177 		// r.height =
178 		// sty.getActualHeightScale()*(fnt.characterSetMaxY-fnt.characterSetMinY);
179 		// System.out.println(
180 		// text+" : "+r+" -> "+fnt.characterSetMinY+","+fnt.characterSetMaxY );
181 	}

183 	protected int fontAdjustment(final String fontname) {
184 		int xadjust = 0;
185 
186 		// if we do not have a script type font
187 		if (fontname.indexOf("scri") < 0) {
188 			// if we have a gothic font
189 			if (fontname.indexOf("goth") >= 0) {
190 				xadjust = 2;
191 			} else {
192 				xadjust = 3;
193 			}
194 		}
195 
196 		return xadjust;
197 	}

199 	protected void drawCharacter(final int xp, final int yp, final int rotpx, final int rotpy,
200 			final float width, final float height, final boolean rotate, final float sinTheta,
201 			final float cosTheta, final boolean draw, final char vectors[][],
202 			final int numberOfPoints, final int minX, final int characterSetMinY, final int lineWidth,
203 			final boolean italics, final float slant, final ImageRenderer<T, ?> renderer, final T colour,
204 			final Rectangle bounds)
205 	{
206 		float xd, yd, xd2, yd2;
207 		int oldx = 0, oldy = 0, x, y, i;
208 		boolean skip = true;
209 		final float finalSlant = height * (-slant);
210 		int maxX = 0, maxY = 0;
211 
212 		// loop through each vertex in the character
213 		for (i = 1; i < numberOfPoints; i++) {
214 			// if this is a "skip"
215 			if (vectors[HersheyFontData.X][i] == ' ') {
216 				// set the skip flag
217 				skip = true;
218 			} else {
219 				// calculate italics offset if necessary
220 				x = (int) ((italics) ? ((vectors[HersheyFontData.Y][i] - characterSetMinY) * finalSlant)
221 						: 0)
222 						+
223 						// add italics offset to the "normal" point
224 						// transformation
225 						this.transformX(xp, vectors[HersheyFontData.X][i], minX, width);
226 
227 				// calculate the y coordinate
228 				y = this.transformY(yp, vectors[HersheyFontData.Y][i], characterSetMinY, height);
229 
230 				// if we are doing a rotation
231 				if (rotate) {
232 					// apply the rotation matrix ...
233 
234 					// transform the coordinate to the rotation center point
235 					xd = (x - rotpx);
236 					yd = (y - rotpy);
237 
238 					// rotate
239 					xd2 = xd * cosTheta - yd * sinTheta;
240 					yd2 = xd * sinTheta + yd * cosTheta;
241 
242 					// transform back
243 					x = (int) (xd2 + 0.5) + rotpx;
244 					y = (int) (yd2 + 0.5) + rotpy;
245 				}
246 
247 				if (!skip) {
248 					// if we are to draw the string
249 					if (draw) {
250 						renderer.drawLine(oldx, oldy, x, y, lineWidth, colour);
251 					}
252 
253 					bounds.x = Math.min(bounds.x, x);
254 					bounds.y = Math.min(bounds.y, y);
255 					maxX = Math.max(maxX, x);
256 					maxY = Math.max(maxY, y);
257 				} // end if not skip
258 
259 				skip = false;
260 
261 				oldx = x;
262 				oldy = y;
263 			} // end if skip
264 		} // end for each vertex in the character
265 
266 		bounds.width = Math.max(bounds.width, maxX - bounds.x);
267 		bounds.height = Math.max(bounds.height, maxY - bounds.y);
268 	}

270 	protected final int transformX(final int xoffset, final int px, final int minx, final float mag) {
271 		return ((int) (xoffset + (px - minx) * mag));
272 	}
273 
274 	protected final int transformY(final int yoffset, final int py, final int miny, final float mag) {
275 		return ((int) (yoffset + (py - miny) * mag));
276 	}
