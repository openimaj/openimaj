46  /**
47   * @param <T> 
48   *	
49   */
50  public class MathMLFontStyle<T> extends FontStyle<T>
51  {
52  	enum MathInput{
53  		MATHML,LATEX;
54  	}
55  
56  	/**
57  	 * 
58  	 */
59  	public static final Attribute TEXT_MODE = null;
60  
61  	private MathInput mathInput = MathInput.LATEX;
62  
63  	private boolean textMode;
64  	protected MathMLFontStyle(MathMLFont font, T col) {
65  		super(font, col);
66  	}
67  
68  	/**
69  	 * @return the {@link LayoutContext} encoding of this {@link MathMLFontStyle}
70  	 */
71  	public LayoutContext getLayoutContext() {
72  		MutableLayoutContext mlc = new LayoutContextImpl(
73                  LayoutContextImpl.getDefaultLayoutContext());
74          T col = this.colour;
75          if(col instanceof Float[]){
76          	Float[] mcol = (Float[]) col;
77          	mlc.setParameter(Parameter.MATHCOLOR, new Color(mcol[0], mcol[1], mcol[2]));
78          }
79          else if(col instanceof Float){
80          	mlc.setParameter(Parameter.MATHCOLOR, new Color((Float)col,(Float)col,(Float)col));
81          }
82          mlc.setParameter(Parameter.MATHSIZE, this.getFontSize());
83  		return mlc;
84  	}
85  
86  	/**
87  	 * @return how to interruprate math text
88  	 */
89  	public MathInput getMathInput() {
90  		return this.mathInput ;
91  	}
92  	
93  	/**
94  	 * @param input
95  	 */
96  	public void setMathInput(MathInput input){
97  		this.mathInput = input;
98  	}
99  
100 	/**
101 	 * @return whether text
102 	 */
103 	public boolean isTextMode() {
104 		return this.textMode;
105 	}
106 	
107 	@Override
108 	public void parseAttributes(Map<? extends Attribute, Object> attrs) {
109 		super.parseAttributes(attrs);
110 		if (attrs.containsKey(MathMLFontStyle.TEXT_MODE)) this.textMode = ((Boolean) attrs.get(TEXT_MODE)).booleanValue();
111 		
112 	}
