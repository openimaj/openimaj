40  /**
41   * A Sparse array of int values implemented using a pair of 
42   * sorted parallel arrays. The first array holds the indexes of
43   * the non-zero elements in ascending order; the second holds
44   * the corresponding values.
45   * <p>
46   * Element access has worst-case O(log n) performance. Appends (where
47   * the index being added is greater than the existing indices) has O(1)
48   * complexity. Random puts have worst case O(n + log n) complexity. 
49   * Complexity for getting values by increasing index for iteration 
50   * through the non-zero values is O(1).
51   * <p>
52   * In summary, this implementation has good access performance, and is
53   * fast for appending values, but slow for putting at random indices. 
54   * It is is optimal for a scenario where you first create the sparse 
55   * array and put values in order of increasing indices, and later use 
56   * the array mostly for reading.
57   *  
58   * @see "http://stackoverflow.com/questions/1880804/java-time-efficient-sparse-1d-array-double"
59   * 
60   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
61   *
62   */

294 	private int update(int spot, int key, int value) {
295 		// grow if reaching end of capacity
296 		if (used == keys.length) {
297 			int capacity = (keys.length * 3) / 2 + 1;
298 			keys = Arrays.copyOf(keys, capacity);
299 			values = Arrays.copyOf(values, capacity);
300 		}
301 		
302 		// shift values if not appending
303 		if (spot < used) {
304 			System.arraycopy(keys, spot, keys, spot + 1, used - spot);
305 			System.arraycopy(values, spot, values, spot + 1, used - spot);
306 		}
307 		
308 		used++;
309 		keys[spot] = key;
310 		return values[spot] = value;
311 	}
