30  package org.openimaj.aop;

32  import java.io.ByteArrayInputStream;
33  import java.lang.instrument.ClassFileTransformer;
34  import java.lang.instrument.IllegalClassFormatException;
35  import java.security.ProtectionDomain;
36  import java.util.ArrayList;
37  import java.util.List;
38  
39  import org.apache.logging.log4j.Logger;
40  import org.apache.logging.log4j.LogManager;
41  
42  import javassist.CannotCompileException;
43  import javassist.ClassPool;
44  import javassist.CtClass;
45  import javassist.NotFoundException;
46  import javassist.Translator;
47  
48  /**
49   * A {@link ClassFileTransformer} that applies one or more
50   * {@link ClassTransformer}s to a class before it is loaded.
51   *
52   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
53   */
54  public class MultiTransformClassFileTransformer implements ClassFileTransformer, Translator {
55  	private static Logger logger = LogManager.getLogger(MultiTransformClassFileTransformer.class);
56  
57  	private ClassPool classPool;
58  	private List<ClassTransformer> transformers = new ArrayList<ClassTransformer>();
59  
60  	/**
61  	 * Construct with the given {@link ClassTransformer}s.
62  	 *
63  	 * @param t1
64  	 *            the first transformer
65  	 * @param transformers
66  	 *            any additional transformers
67  	 */
68  	public MultiTransformClassFileTransformer(ClassTransformer t1, ClassTransformer... transformers) {
69  		this.transformers.add(t1);
70  
71  		for (final ClassTransformer ct : transformers)
72  			this.transformers.add(ct);
73  	}
74  
75  	@Override
76  	public byte[] transform(ClassLoader loader, String className, Class<?> classBeingRedefined,
77  			ProtectionDomain protectionDomain, byte[] classfileBuffer) throws IllegalClassFormatException
78  	{
79  		return transform(className, classfileBuffer);
80  	}
81  
82  	/**
83  	 * Transform the given class.
84  	 *
85  	 * @param className
86  	 *            the name of the class
87  	 * @param classfileBuffer
88  	 *            the class bytes
89  	 * @return the transformed bytes
90  	 */
91  	public byte[] transform(String className, byte[] classfileBuffer) {
92  		if (classPool == null)
93  			classPool = ClassPool.getDefault();
94  
95  		try {
96  			final CtClass ctclz = classPool.makeClass(new ByteArrayInputStream(classfileBuffer));
97  
98  			transform(className, ctclz);
99  
100 			return ctclz.toBytecode();
101 		} catch (final Exception e) {
102 			e.printStackTrace();
103 			logger.error("Error transforming class " + className);
104 			return classfileBuffer;
105 		}
106 	}
107 
108 	private void transform(String className, CtClass ctclz) throws Exception {
109 		for (final ClassTransformer ct : transformers)
110 			ct.transform(className, ctclz);
111 	}
112 
113 	@Override
114 	public void start(ClassPool pool) throws NotFoundException, CannotCompileException {
115 		this.classPool = pool;
116 	}
117 
118 	@Override
119 	public void onLoad(ClassPool pool, String classname) throws NotFoundException, CannotCompileException {
120 		if (classname.endsWith(".package-info"))
121 			return;
122 
123 		final CtClass clz = pool.get(classname);
124 
125 		try {
126 			transform(classname, clz);
127 		} catch (final Exception e) {
128 			throw new CannotCompileException(e);
129 		}
130 	}
131 }
