30  package org.openimaj.stream.provider.twitter;

32  import org.openimaj.util.api.auth.common.TwitterAPIToken;
33  import org.openimaj.util.concurrent.ArrayBlockingDroppingQueue;
34  import org.openimaj.util.concurrent.BlockingDroppingQueue;
35  
36  import twitter4j.Query;
37  import twitter4j.Status;
38  
39  /**
40   * A concrete version of the {@link AbstractTwitterSearchDataset} which pushes
41   * the {@link Status}s into the stream.
42   * 
43   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
44   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
45   */
46  public class TwitterSearchDataset extends AbstractTwitterSearchDataset<Status> {
47  	/**
48  	 * Construct the dataset from the given API token and query. The stream is
49  	 * backed by an {@link ArrayBlockingDroppingQueue} with a single item
50  	 * capacity.
51  	 * 
52  	 * @param query
53  	 *            the Twitter search query
54  	 * @param token
55  	 *            the Twitter api authentication credentials
56  	 */
57  	public TwitterSearchDataset(Query query, final TwitterAPIToken token) {
58  		this(query, token, new ArrayBlockingDroppingQueue<Status>(1));
59  	}
60  
61  	/**
62  	 * Construct the dataset from the given API token, query and buffer.
63  	 * 
64  	 * @param query
65  	 *            the Twitter search query
66  	 * @param token
67  	 *            the Twitter api authentication credentials
68  	 * @param buffer
69  	 *            the buffer to hold {@link Status}s before they are consumed.
70  	 */
71  	public TwitterSearchDataset(Query query, final TwitterAPIToken token, BlockingDroppingQueue<Status> buffer) {
72  		super(token, buffer, query);
73  	}
74  
75  	@Override
76  	protected void registerStatus(Query query, Status status, String rawjson) throws InterruptedException {
77  		register(status);
78  	}
79  
80  	@Override
81  	public Query getQuery() {
82  		return this.query;
83  	}
84  }
