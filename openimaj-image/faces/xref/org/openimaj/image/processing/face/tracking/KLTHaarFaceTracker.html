33  package org.openimaj.image.processing.face.tracking;
34  
35  import java.util.ArrayList;
36  import java.util.Iterator;
37  import java.util.List;
38  
39  import org.openimaj.image.FImage;
40  import org.openimaj.image.processing.face.detection.DetectedFace;
41  import org.openimaj.image.processing.face.detection.HaarCascadeDetector;
42  import org.openimaj.math.geometry.shape.Rectangle;
43  import org.openimaj.video.processing.tracking.BasicObjectTracker;
44  import org.openimaj.video.tracking.klt.KLTTracker;
45  
46  /**
47   * A face tracker that uses the {@link HaarCascadeDetector} to detect faces in
48   * the image and then tracks them using the {@link KLTTracker}.
49   *
50   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
51   *
52   * @created 13 Oct 2011
53   */
54  public class KLTHaarFaceTracker implements FaceTracker<FImage> {
55  	/** The face detector used to detect the faces */
56  	private final HaarCascadeDetector faceDetector = new HaarCascadeDetector();
57  
58  	/** A list of trackers that are tracking faces within the image */
59  	private final List<BasicObjectTracker> trackers = new ArrayList<BasicObjectTracker>();
60  
61  	/** The previous frame */
62  	private FImage previousFrame = null;
63  
64  	/** When all faces are lost, the frame is retried */
65  	private boolean retryFrame = false;
66  
67  	/** The number of frames to force a retry */
68  	private int forceRetry = -1;
69  
70  	/** Used for forcing retry */
71  	private int frameCounter = 0;
72  
73  	private final float detectionScalar = 1.2f;
74  
75  	/**
76  	 * Default constructor that takes the minimum size (in pixels) of detections
77  	 * that should be considered faces.
78  	 *
79  	 * @param minSize
80  	 *            The minimum size of face boxes
81  	 */
82  	public KLTHaarFaceTracker(final int minSize) {
83  		this.faceDetector.setMinSize(minSize);
84  	}
85  
86  	/**
87  	 * Used to detect faces when there is no current state.
88  	 *
89  	 * @return The list of detected faces
90  	 */
91  	private List<DetectedFace> detectFaces(final FImage img) {
92  		return this.faceDetector.detectFaces(img);
93  	}
94  
95  	/**
96  	 * {@inheritDoc}
97  	 *
98  	 * @see org.openimaj.image.processing.face.tracking.FaceTracker#trackFace(org.openimaj.image.Image)
99  	 */
100 	@Override
101 	public List<DetectedFace> trackFace(final FImage img)
102 	{
103 		List<DetectedFace> detectedFaces = new ArrayList<DetectedFace>();
104 
105 		// Determine whether we need to force a retry now
106 		if (this.forceRetry != -1 && this.frameCounter % this.forceRetry == 0)
107 			this.trackers.clear();
108 
109 		// If we're just starting tracking, find some features and start
110 		// tracking them.
111 		if (this.previousFrame == null || this.trackers.size() == 0) {
112 
113 			// Detect the faces in the image.
114 			final List<DetectedFace> faces = this.detectFaces(img);
115 
116 			// Create trackers for each face found
117 			for (final DetectedFace face : faces) {
118 				// Create a new tracker for this face
119 				final BasicObjectTracker faceTracker = new BasicObjectTracker();
120 				final Rectangle r = face.getBounds();
121 				r.scaleCentroid(this.detectionScalar);
122 				faceTracker.initialiseTracking(r, img);
123 				this.trackers.add(faceTracker);
124 
125 				// Store the last frame
126 				this.previousFrame = img;
127 			}
128 
129 			detectedFaces = faces;
130 		} else
131 			// If we have a previous frame, attempt to track the frame
132 			if (this.previousFrame != null) {
133 				// Update all the trackers
134 				final Iterator<BasicObjectTracker> i = this.trackers.iterator();
135 				while (i.hasNext()) {
136 					final BasicObjectTracker tracker = i.next();
137 					if (tracker.trackObject(img).size() == 0)
138 						i.remove();
139 					else {
140 						// Store the bounding box of the tracked features as the
141 						// face
142 						detectedFaces
143 						.add(new DetectedFace(
144 								tracker.getFeatureList().getBounds(),
145 								img.extractROI(tracker.getFeatureList().getBounds()),
146 								tracker.getFeatureList().countRemainingFeatures()));
147 
148 						// Store the last frame
149 						this.previousFrame = img;
150 					}
151 				}
152 
153 				if (this.trackers.size() == 0 && this.retryFrame == false) {
154 					this.retryFrame = true;
155 					detectedFaces = this.trackFace(img);
156 				}
157 			}
158 
159 		this.frameCounter++;
160 		this.retryFrame = false;
161 		return detectedFaces;
162 	}
163 
164 	/**
165 	 * @return the forceRetry
166 	 */
167 	public int getForceRetry()
168 	{
169 		return this.forceRetry;
170 	}
171 
172 	/**
173 	 * @param forceRetry
174 	 *            the forceRetry to set
175 	 */
176 	public void setForceRetry(final int forceRetry)
177 	{
178 		this.forceRetry = forceRetry;
179 	}
