30  package org.openimaj.image.processing.morphology;
31  
32  import org.openimaj.image.FImage;
33  import org.openimaj.image.pixel.ConnectedComponent;
34  import org.openimaj.image.processor.ImageProcessor;
35  import org.openimaj.image.processor.connectedcomponent.ConnectedComponentProcessor;
36  
37  /**
38   * Morphological thickening of connected components and (assumed binary) FImages.
39   * 
40   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
41   */
42  public class Thicken implements ConnectedComponentProcessor, ImageProcessor<FImage> {
43  	protected HitAndMiss hitAndMiss;
44  	
45  	/**
46  	 * Construct the thickening operator with the given structuring element
47  	 * @param se the structuring element
48  	 */
49  	public Thicken(StructuringElement... se) {
50  		this.hitAndMiss = new HitAndMiss(se);
51  	}
52  	
53  	@Override
54  	public void process(ConnectedComponent cc) {
55  		hitAndMiss.process(cc);
56  		
57  		cc.getPixels().addAll(hitAndMiss.outputPixels);
58  	}
59  
60  	/* (non-Javadoc)
61  	 * @see org.openimaj.image.processor.ImageProcessor#processImage(org.openimaj.image.Image)
62  	 */
63  	@Override
64  	public void processImage(FImage image) {
65  		FImage newImage = image.process(hitAndMiss, true);
66  		
67  		for (int y=0; y<newImage.height; y++) {
68  			for (int x=0; x<newImage.width; x++) {
69  				if (newImage.pixels[y][x] == 1) {
70  					image.pixels[y][x] = 1;
71  				}
72  			}
73  		}
74  	}
75  }
