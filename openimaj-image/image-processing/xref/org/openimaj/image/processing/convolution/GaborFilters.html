30  package org.openimaj.image.processing.convolution;
31  
32  import org.openimaj.image.FImage;
33  import org.openimaj.util.array.ArrayUtils;
34  
35  /**
36   * Utility methods for creating Gabor Filters
37   * 
38   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
39   * 
40   */
41  public class GaborFilters {
42  	private GaborFilters() {
43  	};
44  
45  	/**
46  	 * Create a jet of (multiscale) Gabor filters in the frequency domain. The
47  	 * returned filters have the highest frequency in the centre; to apply them
48  	 * to an image, use
49  	 * {@link FourierConvolve#convolvePrepared(FImage, FImage, boolean)} with
50  	 * the third argument set to true.
51  	 * 
52  	 * @param width
53  	 *            the width of the image that will be filtered (note that the
54  	 *            returned filters will have twice this width to account of the
55  	 *            imaginary (phase) values [all of which are zero])
56  	 * @param height
57  	 *            the height of the image that will be filtered
58  	 * @param orientationsPerScale
59  	 *            the number of filter orientations for each scale (from HF to
60  	 *            LF)
61  	 * @return the jet of filters
62  	 */
63  	public static FImage[] createGaborJets(int width, int height, int... orientationsPerScale) {
64  		final int nscales = orientationsPerScale.length;
65  		final int nfilters = (int) ArrayUtils.sumValues(orientationsPerScale);
66  
67  		final FImage[] filters = new FImage[nfilters];
68  
69  		final double[][] param = new double[nfilters][];
70  		for (int i = 0, l = 0; i < nscales; i++) {
71  			for (int j = 0; j < orientationsPerScale[i]; j++) {
72  				param[l++] = new double[] {
73  						.35,
74  						.3 / Math.pow(1.85, i),
75  						16.0 * orientationsPerScale[i] * orientationsPerScale[i] / (32.0 * 32.0),
76  						Math.PI / (orientationsPerScale[i]) * j
77  				};
78  			}
79  		}
80  
81  		final double[][] freq = new double[height][width];
82  		final double[][] phase = new double[height][width];
83  
84  		final float hw = width / 2f;
85  		final float hh = height / 2f;
86  
87  		for (int y = 0; y < height; y++) {
88  			for (int x = 0; x < width; x++) {
89  				final float fx = x - hw;
90  				final float fy = y - hh;
91  				freq[y][x] = Math.sqrt(fx * fx + fy * fy);
92  				phase[y][x] = Math.atan2(fy, fx);
93  			}
94  		}
95  
96  		for (int i = 0; i < nfilters; i++) {
97  			filters[i] = new FImage(width * 2, height);
98  			for (int y = 0; y < height; y++) {
99  				for (int x = 0; x < width; x++) {
100 					double tr = phase[y][x] + param[i][3];
101 
102 					if (tr > Math.PI)
103 						tr -= 2 * Math.PI;
104 					else if (tr < -Math.PI)
105 						tr += 2 * Math.PI;
106 
107 					filters[i].pixels[y][x * 2] = (float) Math.exp(-10 * param[i][0] *
108 							(freq[y][x] / width / param[i][1] - 1) * (freq[y][x] / width / param[i][1] - 1)
109 							- 2 * param[i][2] * Math.PI * tr * tr
110 							);
111 				}
112 			}
113 		}
114 
115 		return filters;
116 	}
