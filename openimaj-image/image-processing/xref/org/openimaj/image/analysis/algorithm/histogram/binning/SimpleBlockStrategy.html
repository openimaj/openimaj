30  package org.openimaj.image.analysis.algorithm.histogram.binning;
31  
32  import org.openimaj.image.analysis.algorithm.histogram.WindowedHistogramExtractor;
33  import org.openimaj.image.pixel.sampling.RectangleSampler;
34  import org.openimaj.math.geometry.shape.Rectangle;
35  import org.openimaj.math.statistics.distribution.Histogram;
36  
37  /**
38   * A {@link SpatialBinningStrategy} that extracts histograms from a number of
39   * equally-sized, non-overlapping within the sample region and concatenates them
40   * together. Each sub-histogram is L2 normalised.
41   * 
42   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
43   * 
44   */
45  public class SimpleBlockStrategy implements SpatialBinningStrategy {
46  	int numBlocksX;
47  	int numBlocksY;
48  
49  	/**
50  	 * Construct with the given number of blocks in both the x and y directions.
51  	 * 
52  	 * @param numBlocks
53  	 *            number of blocks in each direction
54  	 */
55  	public SimpleBlockStrategy(int numBlocks) {
56  		this(numBlocks, numBlocks);
57  	}
58  
59  	/**
60  	 * Construct with the given number of blocks in the x and y directions.
61  	 * 
62  	 * @param numBlocksX
63  	 *            number of blocks in the x directions
64  	 * @param numBlocksY
65  	 *            number of blocks in the y directions
66  	 */
67  	public SimpleBlockStrategy(int numBlocksX, int numBlocksY) {
68  		this.numBlocksX = numBlocksX;
69  		this.numBlocksY = numBlocksY;
70  	}
71  
72  	@Override
73  	public Histogram extract(WindowedHistogramExtractor binnedData, Rectangle region, Histogram output) {
74  		final float dx = region.width / numBlocksX;
75  		final float dy = region.height / numBlocksY;
76  		final int blockSize = binnedData.getNumBins();
77  
78  		if (output == null || output.values.length != blockSize * numBlocksX * numBlocksY)
79  			output = new Histogram(blockSize * numBlocksX * numBlocksY);
80  
81  		final RectangleSampler rs = new RectangleSampler(region, dx, dy, dx, dy);
82  		int block = 0;
83  		final Histogram tmp = new Histogram(blockSize);
84  
85  		for (final Rectangle r : rs) {
86  			binnedData.computeHistogram(r, tmp);
87  			tmp.normaliseL2();
88  
89  			System.arraycopy(tmp.values, 0, output.values, blockSize * block, blockSize);
90  			block++;
91  		}
92  
93  		return output;
94  	}
95  }
