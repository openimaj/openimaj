39  /**
40   *	A class for generating X, Y and Z axes in a 3D visualisation and providing
41   *	data unit transformations.
42   *
43   *	// TODO: Need to add getters/setters ***

45   *	@author David Dupplaw (dpd@ecs.soton.ac.uk)
46   *  @created 10 Jul 2013

48  public class AxesRenderer3D implements DataUnitsTransformer<float[], double[], double[]>
49  {
50  	/** The x axis renderer */
51  	private final AxisRenderer3D xAxisRenderer = new AxisRenderer3D();
52  
53  	/** The y axis renderer */
54  	private final AxisRenderer3D yAxisRenderer = new AxisRenderer3D();
55  
56  	/** The z axis renderer */
57  	private final AxisRenderer3D zAxisRenderer = new AxisRenderer3D();
58  
59  	/**
60  	 *
61  	 */
62  	public AxesRenderer3D()
63  	{
64  		this.xAxisRenderer.setGridDirection( -1 );
65  		this.yAxisRenderer.getConfig().setOrientation( new double[] { 90, 0, 0, 1, 180, 1, 0, 0 } );
66  		this.zAxisRenderer.getConfig().setOrientation( new double[] { 90, 0, 1, 0 } );
67  		this.yAxisRenderer.getConfig().getRenderingConfig().setNameOrientation(
68  				new double[] { 90, 0, 0, 1 } );
69  		this.yAxisRenderer.getConfig().getRenderingConfig().setNameDirection( -1 );
70  		this.zAxisRenderer.getConfig().getRenderingConfig().setNameOrientation(
71  				new double[] { 90, 0, 1, 0, -90, 1, 0, 0 } );
72  		this.zAxisRenderer.getConfig().getRenderingConfig().setNameDirection( -1 );
73  		this.xAxisRenderer.getConfig().setName( "x-axis" );
74  		this.yAxisRenderer.getConfig().setName( "y-axis" );
75  		this.zAxisRenderer.getConfig().setName( "z-axis" );
76  
77  		this.setAxisThickness( 4 );
78  		this.setDrawMajorGrid( false );
79  		this.setDrawMinorGrid( false );
80  	}
81
