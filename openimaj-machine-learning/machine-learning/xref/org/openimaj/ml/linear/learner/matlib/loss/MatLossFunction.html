30  package org.openimaj.ml.linear.learner.matlib.loss;
31  
32  import org.openimaj.math.matrix.MatlibMatrixUtils;
33  
34  import ch.akuhn.matrix.Matrix;
35  import ch.akuhn.matrix.SparseMatrix;
36  
37  
38  public class MatLossFunction extends LossFunction{
39  	
40  	private LossFunction f;
41  	public MatLossFunction(LossFunction f) {
42  		this.f = f;
43  	}
44  	
45  	@Override
46  	public void setX(Matrix X) {
47  		super.setX(X);
48  		f.setX(X);
49  	}
50  	
51  	@Override
52  	public void setY(Matrix Y) {
53  		super.setY(Y);
54  		f.setY(Y);
55  	}
56  	
57  	@Override
58  	public void setBias(Matrix bias) {
59  		super.setBias(bias);
60  		f.setBias(bias);
61  	}
62  	@Override
63  	public Matrix gradient(Matrix W) {
64  		SparseMatrix ret = SparseMatrix.sparse(W.rowCount(), W.columnCount());
65  		int allRowsY = Y.rowCount()-1;
66  		int allRowsW = W.rowCount()-1;
67  		for (int i = 0; i < Y.columnCount(); i++) {
68  			this.f.setY(MatlibMatrixUtils.subMatrix(Y, 0, allRowsY, i, i));
69  			if(bias!=null) this.f.setBias(MatlibMatrixUtils.subMatrix(bias, 0, allRowsY, i, i));
70  			Matrix w = MatlibMatrixUtils.subMatrix(W, 0, allRowsW, i, i);
71  			Matrix submatrix = f.gradient(w);
72  			MatlibMatrixUtils.setSubMatrix(ret, 0, i, submatrix);
73  		}
74  		return ret;
75  	}
76  
77  	@Override
78  	public double eval(Matrix W) {
79  		double total = 0;
80  		f.setBias(this.bias);
81  		total += f.eval(W);
82  		return total;
83  	}
84  
85  	@Override
86  	public boolean isMatrixLoss() {
87  		return true;
88  	}
89  
90  }
