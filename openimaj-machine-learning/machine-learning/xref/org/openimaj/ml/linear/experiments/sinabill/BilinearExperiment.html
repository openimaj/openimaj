32  import java.io.DataOutputStream;
33  import java.io.File;
34  import java.io.FileOutputStream;
35  import java.io.IOException;

37  import org.apache.log4j.ConsoleAppender;
38  import org.apache.log4j.FileAppender;
39  import org.apache.log4j.Level;
40  import org.apache.log4j.Logger;
41  import org.apache.log4j.PatternLayout;
42  import org.openimaj.io.IOUtils;
43  import org.openimaj.ml.linear.learner.BilinearLearnerParameters;

45  public abstract class BilinearExperiment {
46  	private static final String EXPERIMENT_NAME = "%s/%s/%s_%s";
47  	private static final String PARAMS_NAME = ".paramsascii";
48  	private static final String PARAMS_DATA_NAME = ".params";
49  
50  	private static final String BILL_DATA_ROOT = "%s/TrendMiner/deliverables/year2-18month/Austrian Data/";
51  	private static final String BILL_DATA = "%s/data.mat";
52  	
53  	Logger logger = Logger.getLogger(getClass());
54  	private static long experimentTime = -1;
55  	
56  	protected void prepareExperimentLog(BilinearLearnerParameters params) throws IOException {
57  		ConsoleAppender console = new ConsoleAppender(); //create appender
58  		//configure the appender
59  		String PATTERN = "[%p->%C{1}] %m%n";
60  		console.setLayout(new PatternLayout(PATTERN)); 
61  		console.setThreshold(Level.DEBUG);
62  		console.activateOptions();
63  	  	// add appender to any Logger (here is root)
64  		Logger.getRootLogger().addAppender(console);
65  		File expRoot = prepareExperimentRoot();
66  		
67  		IOUtils.write(params, new DataOutputStream(new FileOutputStream(new File(expRoot,PARAMS_DATA_NAME))));
68  		IOUtils.writeASCII(new File(expRoot,PARAMS_NAME), params);
69  		
70  		File logFile = new File(expRoot,"log");
71  		if(logFile.exists())logFile.delete();
72  		FileAppender file = new FileAppender(new PatternLayout(PATTERN), logFile.getAbsolutePath()); 
73  		file.setThreshold(Level.DEBUG);
74  		file.activateOptions();
75  		Logger.getRootLogger().addAppender(file );
76  		
77  	}
78  	
79  	public File prepareExperimentRoot() throws IOException {
80  		String experimentRoot = String.format(EXPERIMENT_NAME,DATA_ROOT(),getExperimentSetName(),getExperimentName(),""+currentExperimentTime());
81  		File expRoot = new File(experimentRoot);
82  		if(expRoot.exists() && expRoot.isDirectory()) return expRoot;
83  		logger.debug("Experiment root: " + expRoot);
84  		if(!expRoot.mkdirs()) throw new IOException("Couldn't prepare experiment output");
85  		return expRoot;
86  	}
87  
88  	private long currentExperimentTime() {
89  		if(experimentTime==-1){
90  			experimentTime = System.currentTimeMillis();
91  		}
92  		return experimentTime;
93  	}
94  	protected String FOLD_ROOT(int fold) throws IOException {
95  		File foldRoot = new File(prepareExperimentRoot(),String.format("fold_%d",fold));
96  		if(!foldRoot.exists() && !foldRoot.mkdirs()) 
97  			throw new IOException("Failed creating the fold directory: " + foldRoot);
98  		return foldRoot.getAbsolutePath();
99  	}
100 	
101 	protected String MATLAB_DATA() {
102 		
103 		return String.format(BILL_DATA,DATA_ROOT());
104 	}
105 	
106 	protected String MATLAB_DATA(String data) {
107 		
108 		return String.format(data,DATA_ROOT());
109 	}
110 	
111 	protected String DATA_ROOT() {
112 		
113 		return String.format(BILL_DATA_ROOT,DROPBOX_HOME());
114 	}
115 	
116 	private String DROPBOX_HOME() {
117 		String home = System.getProperty("user.home");
118 		
119 		return String.format("%s/Dropbox",home);
120 	}
