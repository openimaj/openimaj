38  /**
39   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
40   *
41   * @param <INPUTALL>
42   * @param <INPUTSINGLE>
43   * @param <INPUTTS>
44   * 
45   * Given a double provider, convert and process returning a {@link DoubleTimeSeries}
46   */
47  public class DoubleProviderTimeSeriesConverter<
48  			INPUTALL,
49  			INPUTSINGLE,
50  			INPUTTS extends TimeSeries<INPUTALL,INPUTSINGLE,INPUTTS> & DoubleTimeSeriesProvider 
51  		>
52  		implements
53  		TimeSeriesConverter<INPUTALL, INPUTSINGLE, INPUTTS, double[], Double, DoubleTimeSeries> {
54  
55  	@Override
56  	public DoubleTimeSeries convert(INPUTTS series) {
57  		return series.doubleTimeSeries();
58  	}
59  
60  	@Override
61  	public DoubleTimeSeries convert(INPUTTS series,TimeSeriesProcessor<double[], Double, DoubleTimeSeries> processor) {
62  		DoubleTimeSeries r = series.doubleTimeSeries();
63  		processor.process(r);
64  		return r;
65  	}
