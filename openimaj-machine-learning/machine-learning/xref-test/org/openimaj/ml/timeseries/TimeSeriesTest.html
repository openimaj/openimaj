149 	public void testGenericTimeSeries(){
150 		class ConcreteTimeSeriesString extends ConcreteTimeSeries<String,ConcreteTimeSeriesString>{
151 
152 			@Override
153 			public ConcreteTimeSeriesString newInstance() {
154 				return new ConcreteTimeSeriesString();
155 			}
156 			
157 		}
158 		ConcreteTimeSeries<String,ConcreteTimeSeriesString> ts = new ConcreteTimeSeriesString();
159 		ts.add(1,"One");
160 		ts.add(2,"Two");
161 		ts.add(5,"Five");
162 		ts.add(9,"Nine");
163 		ts.add(10,"Ten");
164 		
165 		assertTrue(ts.get(1).getData()[0].equals("One"));
166 		assertTrue(ts.get(3).size() == 0);
167 		
168 		ConcreteTimeSeriesString get5d = ts.get(5,2,2);
169 		String[] get5 = get5d.getData();
170 		long[] get5l = get5d.getTimes();
171 		assertTrue(Arrays.equals(get5l, new long[]{1,2,5,9,10}));
172 		assertTrue(get5.length == 5);
173 		assertTrue(get5[0].equals("One"));
174 		assertTrue(get5[1].equals("Two"));
175 		assertTrue(get5[2].equals("Five"));
176 		assertTrue(get5[3].equals("Nine"));
177 		assertTrue(get5[4].equals("Ten"));
178 		
179 		get5 = new String[5];
180 		get5 = ts.get(5,2,2,get5d).getData();
181 		get5l = get5d.getTimes();
182 		assertTrue(Arrays.equals(get5l, new long[]{1,2,5,9,10}));
183 		assertTrue(get5[0].equals("One"));
184 		assertTrue(get5[1].equals("Two"));
185 		assertTrue(get5[2].equals("Five"));
186 		assertTrue(get5[3].equals("Nine"));
187 		assertTrue(get5[4].equals("Ten"));
188 		
189 		ConcreteTimeSeries<String,?> get4d = ts.get(6,2,2);
190 		String[] get4 = get4d.getData();
191 		assertTrue(get4.length == 4);
192 		get5l = get5d.getTimes();
193 		assertTrue(Arrays.equals(get4d.getTimes(), new long[]{2,5,9,10}));
194 		
195 		String[] get3 = ts.get(5,0,2).getData();
196 		assertTrue(get3.length == 3);
197 		assertTrue(get3[0].equals("Five"));
198 		get3 = ts.get(5,2,0).getData();
199 		assertTrue(get3.length == 3);
200 		assertTrue(get3[0].equals("One"));
201 		
202 		String[] get1 = ts.get(5, 1l, 1l).getData();
203 		assertTrue(get1[0].equals("Five"));
204 		assertTrue(get1.length == 1);
205 		String[] get0 = ts.get(7, 1l, 1l).getData();
206 		assertTrue(get0.length == 0);
207 		get1 = ts.get(8, 1l, 1l).getData();
208 		assertTrue(get1.length == 1);
209 		assertTrue(get1[0].equals("Nine"));
210 		
211 		get1 = ts.get(3, 1l, 1l).getData();
212 		assertTrue(get1.length == 1);
213 		assertTrue(get1[0].equals("Two"));
214 		ConcreteTimeSeries<String,?> get2d = ts.get(2, 1l, 1l);
215 		String[] get2 = get2d.getData();
216 		assertTrue(Arrays.equals(get2d.getTimes(), new long[]{1,2}));
217 		assertTrue(get2.length == 2);
