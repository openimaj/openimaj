150 		int[][] newSourceOneDocument = RandomData.getRandomIntArray(10, 100, bottomSmall, topSmall,randomSeed+1);
151 		int[][] newSourceTwoDocument = RandomData.getRandomIntArray(10, 100, bottomBig, topBig,randomSeed+1);
152 		
153 		int[] newClusterSourceOne = rdf.assign(newSourceOneDocument);
154 		int[] newClusterSourceTwo = rdf.assign(newSourceTwoDocument);
155 		
156 		double scoreOneVsOne = scoreSharedTerms(newClusterSourceOne,clusterSourceOne);
157 		double scoreOneVsTwo = scoreSharedTerms(newClusterSourceOne,clusterSourceTwo);
158 		
159 		double scoreTwoVsOne = scoreSharedTerms(newClusterSourceTwo,clusterSourceOne);
160 		double scoreTwoVsTwo = scoreSharedTerms(newClusterSourceTwo,clusterSourceTwo);
161 		
162 		assertTrue(scoreOneVsOne > scoreOneVsTwo);
163 		assertTrue(scoreTwoVsTwo > scoreTwoVsOne);
164 	}

167 	private double scoreSharedTerms(int[] one, int[] two) {
168 		Map<Integer,Double> termFreqOne = termFreq(one);
169 		Map<Integer,Double> termFreqTwo = termFreq(two);
170 		
171 		Set<Integer> sharedKeys = termFreqOne.keySet();
172 		sharedKeys.retainAll(termFreqTwo.keySet());
173 		double score = 0;
174 		for(int key : sharedKeys){
175 			double qtf = Math.abs(termFreqOne.get(key));
176 			double dtf = Math.abs(termFreqTwo.get(key));
177 			score += -1 * (Math.abs(qtf - dtf) - qtf - dtf);
178 		}
179 		return score;
180 	}

183 	private Map<Integer, Double> termFreq(int[] two) {
184 		Map<Integer,Double> out = new HashMap<Integer,Double>();
185 		for(int i : two){
186 			if(!out.containsKey(i))
187 				out.put(i, 0d);
188 			out.put(i, out.get(i)+1);
189 		}
190 		for(int i : out.keySet()){
191 			out.put(i, out.get(i) / two.length);
192 		}
193 		return out;
194 	}
