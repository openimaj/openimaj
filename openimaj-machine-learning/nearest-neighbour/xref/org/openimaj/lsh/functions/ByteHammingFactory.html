63  	private class Function extends ByteHashFunction {
64  		private int ham;
65  
66  		Function(ByteHammingFactory options, int ndims, MersenneTwister rng) {
67  			super(rng);
68  
69  			Uniform uniform = new Uniform(rng);
70  			
71  			if (options.bitsPerDim == 0)
72  				ham = (int) uniform.nextIntFromTo(0, ndims - 1);
73  			else
74  				ham = (int) uniform.nextIntFromTo(0, (ndims * options.bitsPerDim) - 1);
75  		}
76  
77  		@Override
78  		public int computeHashCode(byte[] point) {
79  			// which hash function
80  			if (bitsPerDim == 0) {
81  				return point[ham] == 0 ? 0 : 1;
82  			} else {
83  				// compact binary data
84  				final int m = ham % bitsPerDim;
85  				final int d = ham / bitsPerDim;
86  				return (int) (HammingHelper.convert(point[d]) >>> m & 1L);
87  			}
88  		}
89  		
90  		@Override
91  		public int computeHashCode(SparseByteArray array) {
92  			// which hash function
93  			if (bitsPerDim == 0) {
94  				return array.get(ham) == 0 ? 0 : 1;
95  			} else {
96  				// compact binary data
97  				final int m = ham % bitsPerDim;
98  				final int d = ham / bitsPerDim;
99  				return (int) (HammingHelper.convert(array.get(d)) >>> m & 1L);
100 			}			
101 		}
102 	}
