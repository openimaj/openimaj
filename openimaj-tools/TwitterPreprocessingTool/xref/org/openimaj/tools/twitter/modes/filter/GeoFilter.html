30  package org.openimaj.tools.twitter.modes.filter;

32  import java.util.ArrayList;
33  import java.util.List;
34  import org.kohsuke.args4j.Option;
35  import org.openimaj.math.geometry.point.Point2d;
36  import org.openimaj.math.geometry.point.Point2dImpl;
37  import org.openimaj.math.geometry.shape.Rectangle;
38  import org.openimaj.twitter.USMFStatus;

40  /**
41   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
42   *
43   */
44  public class GeoFilter extends TwitterPreprocessingPredicate {
45  	
46  	@Option(name="--bounding-box", aliases="-bb", required=false, usage="The bounding box to filter against.", metaVar="STRING")
47  	String boundBox = "51.28,-0.489,51.686,0.236";
48  	private Rectangle location = null;
49  	
50  	public GeoFilter(){
51  		
52  	}
53  	
54  	@Override
55  	public boolean test(USMFStatus twitterStatus) {
56  		if(twitterStatus.geo == null){
57  			return false;
58  		}
59  		
60  		
61  		//if(geomap == null) geomap = (Map<String, Object>) twitterStatus.coordinates;
62  //		"geo":{"type":"Point","coordinates":[51.55047862,-0.29938507]}
63  		List<Double> geolist = new ArrayList<Double>();
64  		geolist.add(twitterStatus.geo[0]);
65  		geolist.add(twitterStatus.geo[1]);
66  		Point2d pos = new Point2dImpl(geolist.get(1).floatValue(),geolist.get(0).floatValue());
67  		return location.isInside(pos);
68  	}
69  	
70  	@Override
71  	public void validate() {
72  		getLocation();
73  	}
74  	
75  	private Rectangle getLocation() {
76  		if(this.location == null){
77  			double minLon,minLat,maxLon,maxLat;
78  			String[] parts = boundBox.split(",");
79  			minLat = Double.parseDouble(parts[0]);
80  			minLon = Double.parseDouble(parts[1]);
81  			maxLat = Double.parseDouble(parts[2]);
82  			maxLon = Double.parseDouble(parts[3]);
83  			
84  			Point2d minp = new Point2dImpl((float)minLon,(float)minLat);
85  			Point2d maxp = new Point2dImpl((float)maxLon,(float)maxLat);
86  			this.location  = new Rectangle(minp,maxp);
87  		}
88  		return location;
89  	}
90  }
