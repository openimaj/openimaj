122 	private File fileFromStream(InputStream stream) throws IOException {
123 		File f = folder.newFile("tweet" + stream.hashCode() + ".txt");
124 		PrintWriter writer = new PrintWriter(f,"UTF-8");
125 		BufferedReader reader = new BufferedReader(new InputStreamReader(stream, "UTF-8"));
126 		String line = null;
127 		while((line = reader.readLine()) != null){
128 			writer.println(line);
129 		}
130 		writer.flush(); writer.close();
131 		return f;
132 	}

519 	int[] range(int start, int stop)
520 	{
521 	   int[] result = new int[stop-start];
522 
523 	   for(int i=0;i<stop-start;i++)
524 	      result[i] = start+i;
525 
526 	   return result;
527 	}

530 		return checkSameAnalysis(unanalysed,analysed, m,USMFStatus.class);
531 	}
532 	boolean checkSameAnalysis(File unanalysed,File analysed, TwitterPreprocessingMode<?> m, Class<? extends GeneralJSON> readclass) throws IOException {
533 		return checkSameAnalysis(unanalysed,analysed,m,readclass,GeneralJSONTwitter.class);
534 	}
535 	boolean checkSameAnalysis(File unanalysed,File analysed, TwitterPreprocessingMode<?> m, Class<? extends GeneralJSON> readclass,Class<? extends GeneralJSON> originalClass) throws IOException {
536 		TwitterStatusList<USMFStatus>  unanalysedTweetsF = FileTwitterStatusList.readUSMF(unanalysed,"UTF-8",originalClass);
537 		TwitterStatusList<USMFStatus>  analysedTweetsF = FileTwitterStatusList.readUSMF(analysed,"UTF-8",readclass);
538 
539 		MemoryTwitterStatusList<USMFStatus> unanalysedTweets = new MemoryTwitterStatusList<USMFStatus>();
540 		for (USMFStatus twitterStatus : unanalysedTweetsF) {
541 			if(twitterStatus.isInvalid()) continue;
