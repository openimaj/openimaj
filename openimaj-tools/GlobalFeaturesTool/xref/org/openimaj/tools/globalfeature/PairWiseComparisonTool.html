54  /**
55   * A tool for computing the similarities/distances between two images based
56   * on a global feature.
57   * 
58   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
59   *
60   */
61  public class PairWiseComparisonTool {
62  	@Option(name="--image-1", aliases="-im1", usage="first image", handler=MBFImageOptionHandler.class, required=true)
63  	private MBFImage im1;
64  	
65  	@Option(name="--image-2", aliases="-im2", usage="second image", handler=MBFImageOptionHandler.class, required=true)
66  	private MBFImage im2;
67  	
68  	@Option(name="--metric", aliases="-m", usage="comparison metric", required=true)
69  	private FeatureComparison compare;
70  	
71  	@Option(name="--feature-type", aliases="-f", handler=ProxyOptionHandler.class, usage="Feature type", required=true)
72  	private GlobalFeatureType feature;
73  	private GlobalFeatureExtractor featureOp;
74  	
75  	@SuppressWarnings("unchecked")
76  	protected <T extends FeatureVector> FVComparator<T> getComp(T fv, FeatureComparison type) {
77  		if (fv instanceof ByteFV) return (FVComparator<T>) ByteFVComparison.valueOf(type.name());
78  		if (fv instanceof ShortFV) return (FVComparator<T>) ShortFVComparison.valueOf(type.name());
79  		if (fv instanceof IntFV) return (FVComparator<T>) IntFVComparison.valueOf(type.name());
80  		if (fv instanceof FloatFV) return (FVComparator<T>) FloatFVComparison.valueOf(type.name());
81  		if (fv instanceof DoubleFV) return (FVComparator<T>) DoubleFVComparison.valueOf(type.name());
82  		return null;
83  	}
84  	
85  	double execute() {
86  		FeatureVector fv1 = featureOp.extract(im1);
87  		FeatureVector fv2 = featureOp.extract(im2);
88  		
89  		if (compare == FeatureComparison.EQUALS) {
90  			if (Arrays.equals(fv1.asDoubleVector(), fv2.asDoubleVector()))
91  				return 1;
92  			return 0;
93  		} else {
94  			return getComp(fv1, compare).compare(fv1, fv2);
95  		}
96  	}
97  	
98  	/**
99  	 * The main method of the tool.
100 	 * @param args
101 	 */
102 	public static void main(String [] args) {
103 		PairWiseComparisonTool tool = new PairWiseComparisonTool();
104 		
105 		CmdLineParser parser = new CmdLineParser(tool);
106 
107 		try {
108 			parser.parseArgument(args);
109 		} catch (CmdLineException e) {
110 			System.err.println(e.getMessage());
111 			System.err.println("Usage: pairwisecomp [options...]");
112 			parser.printUsage(System.err);
113 			
114 			if (tool.feature == null) {
115 				for (GlobalFeatureType m : GlobalFeatureType.values()) {
116 					System.err.println();
117 					System.err.println(m + " options: ");
118 					new CmdLineParser(m.getOptions()).printUsage(System.err);
119 				}
120 			}
121 			return;
122 		}
123 		
124 		System.out.println(tool.execute());
125 	}
