30  package org.openimaj.tools.faces.recognition;

32  import java.io.File;
33  import java.io.IOException;
34  import java.util.List;
35  
36  import org.kohsuke.args4j.CmdLineException;
37  import org.kohsuke.args4j.CmdLineParser;
38  import org.openimaj.image.ImageUtilities;
39  import org.openimaj.image.processing.face.detection.DetectedFace;
40  import org.openimaj.image.processing.face.recognition.FaceRecognitionEngine;
41  import org.openimaj.ml.annotation.ScoredAnnotation;
42  import org.openimaj.util.pair.IndependentPair;
43  
44  /**
45   * A tool for using pre-trained face recognisers
46   * 
47   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
48   */
49  public class FaceRecognitionTool {
50  
51  	/**
52  	 * The main method of the tool.
53  	 * 
54  	 * @param <T>
55  	 *            Type of DetectedFace
56  	 * 
57  	 * @param args
58  	 * @throws IOException
59  	 * @throws ClassNotFoundException
60  	 */
61  	public static <T extends DetectedFace> void main(String[] args) throws IOException, ClassNotFoundException {
62  		final FaceRecognitionToolOptions options = new FaceRecognitionToolOptions();
63  		final CmdLineParser parser = new CmdLineParser(options);
64  
65  		try {
66  			parser.parseArgument(args);
67  		} catch (final CmdLineException e) {
68  			System.err.println(e.getMessage());
69  			System.err.println("java FaceRecogniserTrainingTool [options...] IMAGE-FILES-OR-DIRECTORIES");
70  			parser.printUsage(System.err);
71  			return;
72  		}
73  
74  		final FaceRecognitionEngine<T, String> engine = options.getEngine();
75  
76  		if (options.files == null) {
77  			System.err.println("No images to test");
78  			return;
79  		}
80  
81  		for (final File f : options.files) {
82  			System.out.println(f);
83  
84  			final List<IndependentPair<T, ScoredAnnotation<String>>> res = engine.recogniseBest(ImageUtilities.readF(f));
85  			for (int i = 0; i < res.size(); i++) {
86  				System.out.println("Face " + i + ": " + res.get(i).firstObject().getBounds() + " -> "
87  						+ res.get(i).secondObject());
88  			}
89  
90  			System.out.println();
91  		}
92  	}
93  
94  }
