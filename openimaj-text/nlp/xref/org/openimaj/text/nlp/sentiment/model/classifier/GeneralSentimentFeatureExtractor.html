30  package org.openimaj.text.nlp.sentiment.model.classifier;
31  
32  import java.util.ArrayList;
33  import java.util.HashMap;
34  import java.util.List;
35  import java.util.Map.Entry;
36  
37  import org.openimaj.feature.DoubleFV;
38  import org.openimaj.feature.FeatureExtractor;
39  import org.openimaj.ml.annotation.AbstractAnnotator;
40  import org.openimaj.ml.annotation.bayes.NaiveBayesAnnotator;

42  /**
43   * {@link FeatureExtractor} that is suitable for {@link NaiveBayesAnnotator}.
44   * Should be initialized with training corpus of the machine learning
45   * {@link AbstractAnnotator} you are using.
46   *
47   * @author Laurence Willmore (lgw1e10@ecs.soton.ac.uk)
48   *
49   */
50  public class GeneralSentimentFeatureExtractor implements
51  FeatureExtractor<DoubleFV, List<String>>
52  {
53  
54  	private ArrayList<String> vocabList;
55  	private int wordOccuranceThresh = 50;
56  
57  	/**
58  	 * Construct with the training set. This is required to build a vocabulary.
59  	 *
60  	 * @param domainVocabularyCorpus
61  	 *            list of tokenised corpus documents.
62  	 */
63  	public GeneralSentimentFeatureExtractor(
64  			List<List<String>> domainVocabularyCorpus)
65  	{
66  		initialize(domainVocabularyCorpus);
67  	}
68  
69  	/**
70  	 * Blank constructor. Will require initialize to be called at a later stage.
71  	 */
72  	public GeneralSentimentFeatureExtractor() {
73  
74  	}
75  
76  	/**
77  	 * Allows a new vocabulary to be constructed from a new corpus.
78  	 *
79  	 * @param domainVocabularyCorpus
80  	 *            list of tokenised corpus documents.
81  	 */
82  	public void initialize(List<List<String>> domainVocabularyCorpus) {
83  		final HashMap<String, Integer> vocab = new HashMap<String, Integer>();
84  		for (final List<String> doc : domainVocabularyCorpus) {
85  			for (final String s : doc) {
86  				Integer current = vocab.get(s);
87  				if (current == null)
88  					current = 0;
89  				vocab.put(s, current + 1);
90  
91  			}
92  		}
93  		this.vocabList = new ArrayList<String>();
94  		for (final Entry<String, Integer> entry : vocab.entrySet()) {
95  			if (entry.getValue() > wordOccuranceThresh) {
96  				vocabList.add(entry.getKey());
97  			}
98  		}
99  	}
100 
101 	@Override
102 	public DoubleFV extractFeature(List<String> tokens) {
103 		final double[] vect = new double[vocabList.size()];
104 		for (int i = 0; i < vect.length; i++) {
105 			vect[i] += 0.00001;
106 		}
107 		for (final String s : tokens) {
108 			final int ind = vocabList.indexOf(s);
109 			if (ind >= 0)
110 				vect[ind] += 1;
111 		}
112 		final double[] vectNorm = new double[vocabList.size()];
113 		for (int i = 0; i < vect.length; i++) {
114 			vectNorm[i] = vect[i] / tokens.size();
115 		}
116 		return new DoubleFV(vect);
117 	}
