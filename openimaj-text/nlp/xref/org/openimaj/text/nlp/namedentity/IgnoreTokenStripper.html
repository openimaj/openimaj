32  import java.io.BufferedReader;
33  import java.io.DataInputStream;
34  import java.io.InputStream;
35  import java.io.InputStreamReader;
36  import java.util.ArrayList;
37  import java.util.HashSet;
38  import java.util.List;
39  import java.util.regex.Matcher;
40  import java.util.regex.Pattern;

42  /**
43   * Class to remove stopwords from a list of tokens, or to check if a word is a
44   * stopword.
45   * 
46   * @author Laurence Willmore (lgw1e10@ecs.soton.ac.uk)
47   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
48   */
49  public class IgnoreTokenStripper {
50  	/**
51  	 * Language to build stripper from.
52  	 */
53  	public enum Language {
54  		English
55  	};
56  
57  	private String units = "one|two|three|four|five|six|seven|eight|nine";
58  	private String tens = "twenty|thirty|forty|fifty|sixty|seventy|eighty|ninety";
59  	private String teens = "ten|eleven|twelve|thirteen|fourteen|fifteen|sixteen|seventeen|eighteen|nineteen";
60  	private String and = "\\s*-?\\s*and\\s*-?\\s*";
61  	private String toNN = "[" + units + "|" + teens + "] | [" + tens + "]\\s*-?\\s*[" + units + "]";
62  	private String toNNN = toNN + " | [[" + units + "]\\s*-?\\s*hundred [" + and + "[" + toNN + "]+]+]";
63  	/*
64  	 * This currently recognizes written numbers up to nine hundred and ninety
65  	 * nine.
66  	 */
67  	private Pattern writtenNumbers = Pattern.compile("[" + toNNN + "]+");
68  
69  	private HashSet<String> ignoreTokens;
70  
71  	/**
72  	 * Constructor for specified language.
73  	 * 
74  	 * @param language
75  	 */
76  	public IgnoreTokenStripper(Language language) {
77  		this.ignoreTokens = new HashSet<String>();
78  		for (final InputStream fstream : getListStreams(language)) {
79  			addToIgnoreSet(fstream);
80  		}
81  	}
82  
83  	private void addToIgnoreSet(InputStream fstream) {
84  		try {
85  			final DataInputStream in = new DataInputStream(fstream);
86  			final BufferedReader br = new BufferedReader(new InputStreamReader(in));
87  			String strLine;
88  			// Read File Line By Line
89  			while ((strLine = br.readLine()) != null) {
90  				ignoreTokens.add(strLine.trim());
91  			}
92  			// Close the input stream
93  			in.close();
94  		} catch (final Exception e) {// Catch exception if any
95  			System.err.println("Error: " + e.getMessage());
96  		}
97  	}
98  
99  	private List<InputStream> getListStreams(Language language) {
100 		final ArrayList<InputStream> res = new ArrayList<InputStream>();
101 		if (language.equals(Language.English)) {
102 			res.add(this.getClass().getResourceAsStream(
103 					"/org/openimaj/text/stopwords/en_stopwords.txt"));
104 			res.add(this.getClass().getResourceAsStream(
105 					"/org/openimaj/text/stopwords/en_nouns.txt"));
106 			res.add(this.getClass().getResourceAsStream(
107 					"/org/openimaj/text/stopwords/en_countries.txt"));
108 			return res;
109 		}
110 		else
111 			return null;
112 	}
113 
114 	/**
115 	 * Strips given list of tokens of all ignore words.
116 	 * 
117 	 * @param intokens
118 	 * @return list of clean tokens.
119 	 */
120 	public ArrayList<String> getNonStopWords(List<String> intokens) {
121 		final ArrayList<String> result = new ArrayList<String>();
122 		for (final String string : intokens) {
123 			if (!isIgnoreToken(string)) {
124 				result.add(string);
125 			}
126 		}
127 		return result;
128 	}
129 
130 	/**
131 	 * Checks if given token is an ignore word
132 	 * 
133 	 * @param token
134 	 * @return true if ignore Token
135 	 */
136 	public boolean isIgnoreToken(String token) {
137 		// check if in ignore list
138 		if (ignoreTokens.contains(token))
139 			return true;
140 		// check if it is a number
141 		try {
142 			Double.parseDouble(token);
143 			return true;
144 		} catch (final Exception e) {
145 		}
146 		// check if it is a number written as a word
147 		final Matcher m = writtenNumbers.matcher(token.toLowerCase());
148 		if (m.matches())
149 			return true;
150 		return false;
151 	}
