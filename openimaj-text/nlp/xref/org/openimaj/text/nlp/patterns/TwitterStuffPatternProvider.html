30  package org.openimaj.text.nlp.patterns;

32  import java.util.regex.Pattern;
33  
34  import org.openimaj.text.util.RegexUtil;
35  
36

38  /**
39   * Borrowed heavily from https://github.com/twitter/twitter-text-java
40   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
41   *
42   */
43  public class TwitterStuffPatternProvider extends PatternProvider{
44  
45  	// These constants were lifted directly from the twitter regex class file mentioned above
46  	private static String LATIN_ACCENTS_CHARS = "\\u00c0-\\u00d6\\u00d8-\\u00f6\\u00f8-\\u00ff\\u015f";
47  	private static final String HASHTAG_ALPHA_CHARS = "a-zA-Z" + LATIN_ACCENTS_CHARS +
48      "\\u0400-\\u04ff\\u0500-\\u0527" +  // Cyrillic
49      "\\u2de0-\\u2dff\\ua640-\\ua69f" +  // Cyrillic Extended A/B
50      "\\u1100-\\u11ff\\u3130-\\u3185\\uA960-\\uA97F\\uAC00-\\uD7AF\\uD7B0-\\uD7FF" + // Hangul (Korean)
51      "\\p{InHiragana}\\p{InKatakana}" +  // Japanese Hiragana and Katakana
52      "\\p{InCJKUnifiedIdeographs}" +     // Japanese Kanji / Chinese Han
53      "\\u3005\\u303b" +                  // Kanji/Han iteration marks
54      "\\uff21-\\uff3a\\uff41-\\uff5a" +  // full width Alphabet
55      "\\uff66-\\uff9f" +                 // half width Katakana
56      "\\uffa1-\\uffdc";                  // half width Hangul (Korean)
57  	private static final String HASHTAG_ALPHA_NUMERIC_CHARS = "0-9\\uff10-\\uff19_-" + HASHTAG_ALPHA_CHARS;
58  	private static final String HASHTAG_ALPHA = "[" + HASHTAG_ALPHA_CHARS +"]";
59  	private static final String HASHTAG_ALPHA_NUMERIC = "[" + HASHTAG_ALPHA_NUMERIC_CHARS +"]";
60  	private static String AT_SIGNS_CHARS = "@\uFF20";
61  	private  static final Pattern AT_SIGNS = Pattern.compile("[" + AT_SIGNS_CHARS + "]");
62  
63  
64  	String linkHashtag = "(?:#|\uFF03)(?:" + HASHTAG_ALPHA_NUMERIC + "*" + HASHTAG_ALPHA + HASHTAG_ALPHA_NUMERIC + "*)";
65  	String linkUsernames = "(?:" + AT_SIGNS + "+)([a-z0-9_]{1,20})(/[a-z][a-z0-9_\\-]{0,24})?(?=[^a-zA-Z0-9_])";
66  	String retweet = "(?:(\\b)RT:?(\\b))";
67  
68  	@Override
69  	public String patternString() {
70  		return RegexUtil.regex_or_match(linkUsernames,linkHashtag,retweet);
71  	}
72  
73  	@Override
74  	public Pattern pattern(){
75  		return Pattern.compile(patternString(), Pattern.CASE_INSENSITIVE);
76  	}
77  
78  	/**
79  	 * @return the hashtag component
80  	 */
81  	public String hashtagPatternString() {
82  		return linkHashtag;
83  	}
84  	/**
85  	 * @return the retweet component
86  	 */
87  	public String retweetPatternString() {
88  		return retweet;
89  	}
90  	/**
91  	 * @return the username component
92  	 */
93  	public String usernamePatternString() {
94  		return linkUsernames;
95  	}
96  }
