67  	private String[] longestfirst(String[] emoticons) {
68  		Arrays.sort(emoticons,new Comparator<String>(){
69  
70  			@Override
71  			public int compare(String s1, String s2) {
72  				int s1Longer = s1.length() - s2.length();
73  				if(s1Longer > 0) return -1;
74  				else if(s1Longer < 0) return 1;
75  				else{
76  					return s1.compareTo(s2);
77  				}
78  			}
79  
80  		});
81  		return emoticons;
82  	}
