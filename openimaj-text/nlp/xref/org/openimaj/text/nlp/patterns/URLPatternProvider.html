42  	/**
43  	 * 
44  	 * Implementation of the URL regex from
45  	 * http://daringfireball.net/2010/07/improved_regex_for_matching_urls
46  	 * 
47  	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk), Sina Samangooei
48  	 *         <ss@ecs.soton.ac.uk>
49  	 * 
50  	 */
51  	public static class DFURLPatternProvider extends URLPatternProvider {
52  		/**
53  		 * 
54  		 */
55  		public DFURLPatternProvider() {
56  			Url = "\\b" + "(" + // Capture 1: entire matched URL
57  					"(?:" + "https?://" + // http or https protocol
58  					"|" + // or
59  					"www\\d{0,3}[.]" + // "www.", "www1.", "www2." ... "www999."
60  					"|" + // or
61  					// "([\\S]+[.])+[a-z]{2,4}/" + // looks like domain
62  					// name followed by a slash
63  					"[A-Za-z0-9.\\-]+[.][a-z]{2,4}/" + // looks
64  														// like
65  														// domain
66  														// name
67  														// followed
68  														// by a
69  														// slash
70  					")" + "(?:" + // One or more:
71  					"[^\\s()<>]+" + // Run of non-space, non-()<>
72  					"|" + // or
73  					"\\(([^\\s()<>]+|(\\([^\\s()<>]+\\)))*\\)" + // balanced
74  																	// parens,
75  																	// up to 2
76  																	// levels
77  					")+" + "(?:" + // End with:
78  					"\\(([^\\s()<>]+|(\\([^\\s()<>]+\\)))*\\)" + // balanced
79  																	// parens,
80  																	// up to 2
81  																	// levels
82  					"|" + // or
83  					"[^\\s`!()\\[\\]{};:'\".,<>?\u00AB\u00BB\u201C\u201D\u2018\u2019]" + // not
84  																							// a
85  																							// space
86  																							// or
87  																							// one
88  																							// of
89  																							// these
90  																							// punct
91  																							// chars
92  					")" + ")";
93  		}
94  	}

96  	protected String Url;
97  
98  	/**
99  	 * @param punctuation
100 	 * @param entity
101 	 */
102 	public URLPatternProvider(PunctuationPatternProvider punctuation, EntityPatternProvider entity) {
103 		// final String validLettersAndNumbers = "[a-z\\u00a1-\\uffff0-9]";
104 		final String validLettersAndNumbersAndDots = "[a-z\\u00a1-\\uffff0-9\\-.]";
105 		/*
106 		 * final String hostNamePart = "(?:" + validLettersAndNumbers + "+-?)*"
107 		 * + validLettersAndNumbers + "+"; // something // or //
108 		 * something-something // but // never // just // something-
109 		 */Url = "\\b" +
110 		// protocol identifier
111 				"(?:(?:https?://|ftp://|www\\d{0,3}[.]))" +
112 				// user:pass authentication
113 				"(?:\\S+(?::\\S*)?@)?" + "(?:" +
114 				// IP address exclusion
115 				// private & local networks
116 				"(?!10(?:\\.\\d{1,3}){3})"
117 				//
118 				+ "(?!127(?:\\.\\d{1,3}){3})"
119 				//
120 				+ "(?!169\\.254(?:\\.\\d{1,3}){2})"
121 				//
122 				+ "(?!192\\.168(?:\\.\\d{1,3}){2})"
123 				//
124 				+ "(?!172\\.(?:1[6-9]|2\\d|3[0-1])(?:\\.\\d{1,3}){2})"
125 				//
126 				+
127 				// IP address dotted notation octets
128 				// excludes loopback network 0.0.0.0
129 				// excludes reserved space >= 224.0.0.0
130 				// excludes network & broacast addresses
131 				// (first & last IP address of each class)
132 				"(?:[1-9]\\d?|1\\d\\d|2[01]\\d|22[0-3])"
133 				//
134 				+ "(?:\\.(?:1?\\d{1,2}|2[0-4]\\d|25[0-5])){2}"
135 				//
136 				+ "(?:\\.(?:[1-9]\\d?|1\\d\\d|2[0-4]\\d|25[0-4]))" + "|"
137 				//
138 				+
139 				// // host name
140 				// "(?:"+hostNamePart +")"
141 				// +
142 				// // domain name
143 				// "(?:\\."+hostNamePart+")*"
144 				// +
145 				"(?:" + validLettersAndNumbersAndDots + ")+" +
146 				// TLD identifier
147 				"(?:\\.(?:[a-z\\u00a1-\\uffff]{2,4})" + ")" + ")" +
148 				// port number
149 				"(?::\\d{2,5})?" +
150 				// resource path
151 				"(?:/[^\\s]*)?";
152 	}
