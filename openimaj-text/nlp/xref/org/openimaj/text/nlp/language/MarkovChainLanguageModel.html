42  /**
43   * Code to train, classify and generate language specific text by building a
44   * first order Markov chain.
45   * 
46   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
47   * 
48   */
49  public class MarkovChainLanguageModel {
50  
51  	private Map<Locale, Matrix> chains = new HashMap<Locale, Matrix>();
52  	private Map<Locale, long[]> chainCounts = new HashMap<Locale, long[]>();
53  
54  	/**
55  	 * Generate a new empty markov chain language model
56  	 */
57  	public MarkovChainLanguageModel() {
58  		chains = new HashMap<Locale, Matrix>();
59  		chainCounts = new HashMap<Locale, long[]>();
60  	}
61  
62  	/**
63  	 * 
64  	 * Add an example to a language's markov chain
65  	 * 
66  	 * @param language
67  	 *            the language the example is being added to
68  	 * @param example
69  	 *            the new example to learn from
70  	 * @param encoding
71  	 *            the encoding of the example
72  	 * @throws UnsupportedEncodingException
73  	 */
74  	public void train(Locale language, String example, String encoding) throws UnsupportedEncodingException {
75  		if (!chains.containsKey(language)) {
76  			chains.put(language, new Matrix(256 + 1, 256 + 1));
77  			chainCounts.put(language, new long[256 + 1]);
78  		}
79  
80  		final Matrix chain = chains.get(language);
81  		final long[] chainCount = chainCounts.get(language);
82  		final byte[] data = example.getBytes(encoding);
83  
84  		int currentIndex = 0;
85  		final double[][] chainData = chain.getArray();
86  		for (final byte b : data) {
87  			final int newIndex = (b & 0xff) + 1;
88  			chainData[currentIndex][newIndex] = chainData[currentIndex][newIndex] + 1;
89  			chainCount[currentIndex] += 1;
90  			currentIndex = newIndex;
91  		}
92  
93  	}
94  
95  	/**
96  	 * Train a given language on a stream of text
97  	 * 
98  	 * @param language
99  	 * @param stream
100 	 * @throws IOException
101 	 */
102 	public void train(Locale language, InputStream stream) throws IOException {
103 		if (!chains.containsKey(language)) {
104 			chains.put(language, new Matrix(256 + 1, 256 + 1));
105 			chainCounts.put(language, new long[256 + 1]);
106 		}
107 
108 		final Matrix chain = chains.get(language);
109 		final long[] chainCount = chainCounts.get(language);
110 
111 		int currentIndex = 0;
112 		final double[][] chainData = chain.getArray();
113 		int newIndex = -1;
114 		while ((newIndex = stream.read()) != -1) {
115 			newIndex += 1;
116 			chainData[currentIndex][newIndex] = chainData[currentIndex][newIndex] + 1;
117 			chainCount[currentIndex] += 1;
118 			currentIndex = newIndex;
119 		}
120 	}
121 
122 	/**
123 	 * Generate a string using this model of the desired length
124 	 * 
125 	 * @param language
126 	 * 
127 	 * @param length
128 	 * @param encoding
129 	 * @return the generated string
130 	 * @throws UnsupportedEncodingException
131 	 */
132 	public String generate(Locale language, int length, String encoding) throws UnsupportedEncodingException {
133 
134 		final Matrix chain = this.chains.get(language);
135 		if (chain == null)
136 			return null;
137 		final double[][] chainData = chain.getArray();
138 		final long[] chainCount = this.chainCounts.get(language);
139 
140 		int currentIndex = 0;
141 		final byte[] newString = new byte[length];
142 		final Random r = new Random();
143 		for (int i = 0; i < length; i++) {
144 			final double prob = r.nextDouble();
145 			final double[] currentLine = chainData[currentIndex];
146 			double probSum = 0.0;
147 			int newIndex = 0;
148 			// System.out.println("CURRENT STATE:" + (char)(currentIndex-1));
149 			while (probSum + (currentLine[newIndex] / chainCount[currentIndex]) < prob) {
150 				final double probForIndex = (currentLine[newIndex++] / chainCount[currentIndex]);
151 				// System.out.println(probForIndex);
152 				// if(probForIndex > 0){
153 				// System.out.println("Prob to go to:" + (char)(newIndex-2) +
154 				// " = " + probForIndex);
155 				// }
156 				probSum += probForIndex;
157 			}
158 			// System.out.println("NEW STATE:" + (char)(newIndex-1));
159 			newString[i] = (byte) (newIndex - 1);
160 			currentIndex = newIndex;
161 		}
162 
163 		return new String(newString, encoding);
164 	}
