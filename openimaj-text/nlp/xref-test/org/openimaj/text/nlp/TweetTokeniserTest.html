338 	private void testProvider(PatternProvider provider, IndependentPair<String, Integer>[] pairs) {
339 		final Pattern p = provider.pattern();
340 		for (final IndependentPair<String, Integer> pair : pairs) {
341 			final String string = pair.firstObject();
342 			System.out.println("Original: " + string);
343 			// string = EdgePunctuationPatternProvider.fixedges(string);
344 			// System.out.println(string);
345 			final Matcher matches = p.matcher(string);
346 			final ArrayList<String> allemotes = new ArrayList<String>();
347 			while (matches.find()) {
348 				allemotes.add(string.substring(matches.start(), matches.end()));
349 			}
350 			final String found = StringUtils.join(allemotes, "====");
351 			System.out.format("...%s\n...[%s]\n", string, found);
352 			Assert.assertTrue(allemotes.size() == pair.secondObject());
353 		}
354 	}
