44  /**
45   * A helper class for a common stage type. In this case, a stage that goes from
46   * text to a sequence file of bytes indexed by longs
47   * 
48   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
49   * @param <INPUT_FORMAT>
50   * @param <INPUT_KEY>
51   * @param <INPUT_VALUE>
52   * @param <MAP_OUTPUT_KEY>
53   * @param <MAP_OUTPUT_VALUE>
54   * @param <OUTPUT_KEY>
55   * @param <OUTPUT_VALUE>
56   *
57   */
58  public abstract class MultipleOutputStage<INPUT_FORMAT extends FileInputFormat<INPUT_KEY, INPUT_VALUE>, INPUT_KEY, INPUT_VALUE, MAP_OUTPUT_KEY, MAP_OUTPUT_VALUE, OUTPUT_KEY, OUTPUT_VALUE>
59  		extends
60  			Stage<
61  			INPUT_FORMAT,
62  			FileOutputFormat<OUTPUT_KEY, OUTPUT_VALUE>,
63  			INPUT_KEY, INPUT_VALUE,
64  			MAP_OUTPUT_KEY, MAP_OUTPUT_VALUE,
65  			OUTPUT_KEY, OUTPUT_VALUE
66  			>
67  {
68  
69  	@Override
70  	public abstract Class<? extends MultipleOutputReducer<MAP_OUTPUT_KEY, MAP_OUTPUT_VALUE, OUTPUT_KEY, OUTPUT_VALUE>>
71  			reducer();
72  
73  	@Override
74  	public Job stage(Path[] inputs, Path output, Configuration conf) throws Exception {
75  
76  		final Job job = super.stage(inputs, output, conf);
77  		job.setOutputFormatClass(NullOutputFormat.class);
78  		MultipleOutputs.addNamedOutput(job, "text", TextOutputFormat.class, NullWritable.class, Text.class);
79  		return job;
80  	}
81  }
