32  import java.io.IOException;
33  import java.net.URI;
34  
35  import org.apache.hadoop.conf.Configuration;
36  import org.apache.hadoop.fs.FileSystem;
37  import org.apache.hadoop.fs.LocalFileSystem;
38  import org.apache.hadoop.fs.Path;
39  import org.kohsuke.args4j.CmdLineException;
40  import org.kohsuke.args4j.CmdLineParser;
41  import org.kohsuke.args4j.Option;
42  import org.openimaj.hadoop.sequencefile.SequenceFileUtility;
43  
44  public class HadoopEXIFOptions {
45  	@Option(name="--input", aliases="-i", required=true, usage="Input image FILE.", metaVar="STRING")
46  	private String input;
47  	
48  	@Option(name="--output", aliases="-o", required=true, usage="Output keypoint FILE.", metaVar="STRING")
49  	private String output;
50  	
51  	@Option(name="--output-mode", aliases="-om", required=true, usage="Output Mode.", metaVar="STRING")
52  	private EXIFOutputMode outputMode = EXIFOutputMode.TEXT;
53  
54  	@Option(name="--remove", aliases="-rm", required=false, usage="Remove the existing output location if it exists.", metaVar="BOOLEAN")
55  	private boolean replace = false;
56  	
57  	@Option(name="--exif-path", aliases="-ep", required=false, usage="Path to the exiftools executable", metaVar="STRING")
58  	private String exifPath = "/usr/local/bin/exiftool";
59  	
60  	private String[] args;
61  
62  	private boolean beforeMap;
63  
64  	public HadoopEXIFOptions(String[] args, boolean beforeMap) {
65  		this.args = args;
66  		this.beforeMap = beforeMap;
67  	}
68  
69  	public void prepare() {
70  		CmdLineParser parser = new CmdLineParser(this);
71  		try {
72  			parser.parseArgument(args);
73  			this.validate();
74  		} catch (CmdLineException e) {
75  			System.err.println(e.getMessage());
76  			System.err.println("Usage: java -jar HadoopEXIFTool.jar [options...] [files...]");
77  			parser.printUsage(System.err);
78  			
79  			System.exit(1);
80  		}
81  	}
82  	
83  	private void validate() {
84  		if(replace && beforeMap){
85  			try {
86  				URI outuri = SequenceFileUtility.convertToURI(this.getOutputString());
87  				FileSystem fs = getFileSystem(outuri);
88  				fs.delete(new Path(outuri.toString()), true);
89  			} catch (IOException e) {
90  				
91  			}
92  		}
93  	}
94  	public static FileSystem getFileSystem(URI uri) throws IOException {
95  		Configuration config = new Configuration();
96  		FileSystem fs = FileSystem.get(uri, config);
97  		if (fs instanceof LocalFileSystem) fs = ((LocalFileSystem)fs).getRaw();
98  		return fs;
99  	}
100 
101 	public Path[] getInputPaths() throws IOException {
102 		Path[] sequenceFiles = SequenceFileUtility.getFilePaths(this.getInputString(), "part");
103 		return sequenceFiles;
104 	}
105 
106 	public Path getOutputPath() {
107 		return new Path(SequenceFileUtility.convertToURI(this.getOutputString()).toString());
108 	}
109 	
110 	public String getInputString() {
111 		return input;
112 	}
113 
114 	public String getOutputString() {
115 		return output;
116 	}
117 
118 	public String getExifPath() {
119 		return this.exifPath;
120 	}
121 
122 	public EXIFOutputMode getOutputMode() {
123 		return outputMode;
124 	}
125 }
