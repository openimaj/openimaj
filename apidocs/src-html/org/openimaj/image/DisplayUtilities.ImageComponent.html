184        private static BufferedImage getImage(final JFrame frame)
185        {
186                if (frame == null)
187                        return null;
188
189                if (frame.getContentPane().getComponentCount() > 0
190                                && frame.getContentPane().getComponent(0) instanceof ImageComponent)
191                {
192                        return ((ImageComponent) frame.getContentPane().getComponent(0)).image;
193                }
194
195                return null;
196        }

283         * @return the window

285        public static JFrame createNamedWindow(final String name,
286                        final String title, final boolean autoResize)
287        {
288                if (DisplayUtilities.namedWindows.containsKey(name))
289                        return DisplayUtilities.namedWindows.get(name);
290                final JFrame frame = DisplayUtilities.makeDisplayFrame(title, 0, 0,
291                                null);
292                ((ImageComponent) frame.getContentPane().getComponent(0)).autoResize = autoResize;

294                DisplayUtilities.namedWindows.put(name, frame);
295                return frame;
296        }

629                /**
630                 * Set the image to draw
631                 *
632                 * @param image
633                 *            the image
634                 */
635                public void setImage(final BufferedImage image)
636                {
637                        this.image = image;
638
639                        if (this.autoFit)
640                        {
641                                this.calculateScaleFactorsToFit(image, this.getBounds());
642                        }
643                        else if (this.autoResize)
644                        {
645                                // If the component isn't the right shape, we'll resize the
646                                // component.
647                                if (image.getWidth() != this.getWidth() ||
648                                                image.getHeight() != this.getHeight())
649                                {
650                                        this.setPreferredSize(new Dimension(
651                                                        (int) (image.getWidth() * this.scaleFactorX),
652                                                        (int) (image.getHeight() * this.scaleFactorY)));
653                                        this.setSize(new Dimension(
654                                                        (int) (image.getWidth() * this.scaleFactorX),
655                                                        (int) (image.getHeight() * this.scaleFactorY)));
656                                }
657
658                                final Component c = SwingUtilities.getRoot(this);
659                                if (c == null)
660                                        return;
661                                c.validate();
662
663                                if (c instanceof JFrame && this.autoPack)
664                                {
665                                        final JFrame f = (JFrame) c;
666                                        f.pack();
667                                }
668                        }
669
670                        if (this.showPixelColours)
671                                // This forces a repaint if showPixelColours is true
672                                this.updatePixelColours();
673                        else
674                                this.repaint();
675                }

677                /**
678                 * Given an image, will calculate two scale factors for the X and Y
679                 * dimensions of the image, such that the image will fit within the
680                 * bounds.
681                 *
682                 * @param image
683                 *            The image to fit
684                 * @param bounds
685                 *            The bounds to fit within
686                 */
687                private void calculateScaleFactorsToFit(final BufferedImage image,
688                                final java.awt.Rectangle bounds)
689                {
690                        if (image == null || bounds == null)
691                                return;
692
693                        if (this.autoFit)
694                        {
695                                // If we can stretch the image it's pretty simple.
696                                if (!this.keepAspect)
697                                {
698                                        this.scaleFactorX = bounds.width / (double) image.getWidth();
699                                        this.scaleFactorY = bounds.height / (double) image.getHeight();
700                                }
701                                // Otherwise we need to find the ratios to fit while keeping
702                                // aspect
703                                else
704                                {
705                                        this.scaleFactorX = this.scaleFactorY = Math.min(
706                                                        bounds.width / (double) image.getWidth(),
707                                                        bounds.height / (double) image.getHeight());
708                                }
709                        }
710                }

716                 *            The x image coordinate
717                 * @param y
718                 *            The y image coordinate
719                 */
720                public void moveTo(final double x, final double y)
721                {
722                        if (this.drawX != x || this.drawY != y)
723                        {
724                                this.drawX = x;
725                                this.drawY = y;
726                                this.repaint();
727
728                                for (final ImageComponentListener l : this.listeners)
729                                        l.imagePanned(x, y);
730                        }
731                }

802                /**
803                 * Make sure the x and y position we're drawing the image in is not
804                 * going mad.
805                 */
806                private void sanitiseVars()
807                {
808                        // Make sure we're not going out of the space
809                        // this.moveTo(
810                        // Math.max(
811                        // this.image.getWidth() / this.scaleFactorX / 2,

823                /**
824                 * {@inheritDoc}
825                 *
826                 * @see javax.swing.JComponent#paint(java.awt.Graphics)
827                 */
828                @Override
829                public void paint(final Graphics gfx)
830                {
831                        // Create a double buffer into which we'll draw first.
832                        final BufferedImage img = new BufferedImage(this.getWidth(), this.getHeight(), BufferedImage.TYPE_3BYTE_BGR);
833                        final Graphics2D g = (Graphics2D) img.getGraphics();
834
835                        if (this.drawTransparencyGrid)
836                        {
837                                final BufferedImage transparencyGrid = new BufferedImage(
838                                                this.getWidth(), this.getHeight(),
839                                                BufferedImage.TYPE_3BYTE_BGR);
840                                final Graphics tg = transparencyGrid.getGraphics();
841
842                                final int gridSizeX = (int) (20 * this.scaleFactorX);
843                                final int gridSizeY = (int) (20 * this.scaleFactorY);
844                                for (int y = 0; y < this.getHeight(); y += gridSizeY)
845                                {
846                                        for (int x = 0; x < this.getWidth(); x += gridSizeX)
847                                        {
848                                                final int c = (x / gridSizeX + y / gridSizeY) % 2;
849                                                if (c == 0)
850                                                        tg.setColor(new Color(220, 220, 220));
851                                                else
852                                                        tg.setColor(Color.white);
853
854                                                tg.fillRect(x, y, gridSizeX, gridSizeY);
855                                        }
856                                }
857
858                                g.drawImage(transparencyGrid, 0, 0, null);
859                        }
860
861                        // Draw the image
862                        if (this.image != null)
863                        {
864                                // Scale and translate to the image drawing coordinates
865                                g.scale(this.scaleFactorX, this.scaleFactorY);
866                                g.translate(-this.drawX, -this.drawY);
867
868                                // Blat the image to the screen
869                                g.drawImage(this.image, 0, 0, this.image.getWidth(),
870                                                this.image.getHeight(), null);
871
872                                // Reset the graphics back to the original pixel-based coords
873                                g.translate(this.drawX, this.drawY);
874                                g.scale(1 / this.scaleFactorX, 1 / this.scaleFactorY);
875
876                                // If we're to show pixel colours and we're supposed to do it
877                                // on this time around...
878                                if ((this.showPixelColours || this.showXY)
879                                                && this.drawPixelColour)
880                                {
881                                        final StringBuffer pixelColourStrB = new StringBuffer();
882
883                                        if (this.showXY)
884                                                pixelColourStrB.append("[" + this.pixelX + ","
885                                                                + this.pixelY + "] ");
886
887                                        if (this.showPixelColours)
888                                                pixelColourStrB.append(Arrays
889                                                                .toString(this.currentPixelColour));
890
891                                        // Calculate the size to draw
892                                        final FontMetrics fm = g.getFontMetrics();
893                                        final int fw = fm.stringWidth(pixelColourStrB.toString());
894                                        final int fh = fm.getHeight() + fm.getDescent();
895                                        final int p = 4; // padding
896                                        final int dx = 0;
897                                        int dy = this.getHeight() - (fh + p);
898
899                                        // If the mouse is over where we want to put the box,
900                                        // we'll move the box to another corner
901                                        if (this.mouseX <= dx + fw + p && this.mouseX >= dx &&
902                                                        this.mouseY >= dy && this.mouseY <= dy + fh + p)
903                                                dy = 0;
904
905                                        // Draw a box
906                                        g.setColor(new Color(0, 0, 0, 0.5f));
907                                        g.fillRect(dx, dy, fw + p, fh + p);
908
909                                        // Draw the text
910                                        g.setColor(Color.white);
911                                        g.drawString(pixelColourStrB.toString(), dx + p / 2, dy
912                                                        + fm.getHeight() + p / 2);
913                                }
914                        }
915
916                        // Blat our offscreen image to the screen
917                        gfx.drawImage(img, 0, 0, null);
918
919                        // Store this displayed image
920                        this.displayedImage = img;
921                }

923                /**
924                 * {@inheritDoc}
925                 *
926                 * @see java.awt.event.MouseListener#mouseClicked(java.awt.event.MouseEvent)
927                 */
928                @Override
929                public void mouseClicked(final MouseEvent e)
930                {
931                        if (e.getButton() == MouseEvent.BUTTON1 && this.allowZooming)
932                        {
933                                if (e.isControlDown())
934                                {
935                                        // Scale the scalars down
936                                        this.scaleFactorX /= 2;
937                                        this.scaleFactorY /= 2;
938
939                                        final double moveX = this.drawX - e.getX() / this.scaleFactorX / 2;
940                                        final double moveY = this.drawY - e.getY() / this.scaleFactorY / 2;
941                                        if (this.allowDragging)
942                                                this.moveTo(moveX, moveY);
943                                        else
944                                                this.moveTo(0, 0);
945                                }
946                                else
947                                {
948                                        // Scale the scalars up
949                                        this.scaleFactorX *= 2;
950                                        this.scaleFactorY *= 2;
951
952                                        // Make sure we zoom in on the bit the user clicked on
953                                        if (this.allowDragging)
954                                                this.moveTo(
955                                                                this.drawX + e.getX() / this.scaleFactorX,
956                                                                this.drawY + e.getY() / this.scaleFactorY);
957                                        else
958                                                this.moveTo(0, 0);
959                                }
960
961                                // Make sure we're not going to draw out of bounds.
962                                this.sanitiseVars();
963
964                                this.repaint();
965                        }
966                }

1022                @Override
1023                public void mouseMoved(final MouseEvent e)
1024                {
1025                        if (this.image == null)
1026                                return;
1027
1028                        // Convert the screen coords into image coords
1029                        final double x = e.getX() / this.scaleFactorX + this.drawX;
1030                        final double y = e.getY() / this.scaleFactorY + this.drawY;
1031
1032                        // If we're outside the image we don't print anything
1033                        if (x >= this.image.getWidth() || y >= this.image.getHeight() ||
1034                                        x < 0 || y < 0)
1035                        {
1036                                this.drawPixelColour = false;
1037                                this.repaint();
1038                                return;
1039                        }
1040
1041                        // Pixel coordinates in the image
1042                        this.pixelX = (int) x;
1043                        this.pixelY = (int) y;
1044
1045                        this.mouseX = e.getX();
1046                        this.mouseY = e.getY();
1047
1048                        this.updatePixelColours();

1051                /**
1052                 * Update the display of pixel colours
1053                 */
1054                protected void updatePixelColours()
1055                {
1056                        if (this.showPixelColours && this.image != null)
1057                        {
1058                                // If we don't have the original image, we'll just use the
1059                                // colours from the BufferedImage
1060                                if (this.originalImage == null)
1061                                {
1062                                        final int colour = this.image.getRGB(this.pixelX, this.pixelY);
1063                                        this.currentPixelColour = new Float[3];
1064                                        this.currentPixelColour[0] = (float) ((colour & 0x00ff0000) >> 16);
1065                                        this.currentPixelColour[1] = (float) ((colour & 0x0000ff00) >> 8);
1066                                        this.currentPixelColour[2] = (float) ((colour & 0x000000ff));
1067                                }
1068                                else
1069                                {
1070                                        // If we're outside of the original image's coordinates,
1071                                        // we don't need to do anything else..
1072                                        if (this.pixelX >= this.originalImage.getWidth() || this.pixelX < 0 ||
1073                                                        this.pixelY >= this.originalImage.getHeight() || this.pixelY < 0)
1074                                                return;
1075
1076                                        // If we have the original image we get each of the bands
1077                                        // from it and update the current pixel colour member
1078                                        if (this.originalImage instanceof FImage)
1079                                        {
1080                                                final Object o = this.originalImage.getPixel(this.pixelX, this.pixelY);
1081                                                this.currentPixelColour = new Float[1];
1082                                                this.currentPixelColour[0] = (Float) o;
1083                                        }
1084                                        else if (this.originalImage instanceof MBFImage)
1085                                        {
1086                                                final MBFImage i = (MBFImage) this.originalImage;
1087                                                this.currentPixelColour = new Float[i.numBands()];
1088                                                for (int b = 0; b < i.numBands(); b++)
1089                                                        this.currentPixelColour[b] = i.getBand(b)
1090                                                                        .getPixel(this.pixelX, this.pixelY);
1091                                        }
1092                                }
1093
1094                                this.drawPixelColour = true;
1095                                this.repaint();
1096                        }
1097
1098                        if (this.showXY)
1099                        {
1100                                this.drawPixelColour = true;
1101                                this.repaint();
1102                        }
1103                }

1788         * Display multiple images in an array of frames
1789         *
1790         * @param title

1792         * @param images
1793         *            the images
1794         * @return the frame
1795         */
1796        public static JFrame display(final String title,
1797                        final BufferedImage... images)
1798        {
1799                if (GraphicsEnvironment.isHeadless())
1800                        return null;
1801
1802                final JFrame f = new JFrame(title);
1803
1804                final int box_size = 200;
1805                final int n_images = images.length;
1806                final int n_boxes_x = 4;
1807                final int width = n_boxes_x * box_size;
1808                final int height = box_size * n_images / n_boxes_x;
1809
1810                f.addWindowListener(new WindowAdapter()
1811                {
1812                        @Override
1813                        public void windowClosing(final WindowEvent evt)
1814                        {
1815                                DisplayUtilities.windowOpenCount = DisplayUtilities.windowCount - 1;
1816                                f.dispose();
1817                        }
1818                });
1819
1820                final Container scrollContainer = new Container();

1822
1823                final Container container = new Container();
1824                container.setSize(new Dimension(width, height));
1825                container.setPreferredSize(new Dimension(width, height));
1826                container.setLayout(new GridLayout(0, n_boxes_x));
1827                scrollContainer.add(container);
1828
1829                for (final BufferedImage img : images)
1830                {
1831                        final JComponent c = new JComponent()
1832                        {
1833                                private static final long serialVersionUID = 1L;
1834
1835                                @Override
1836                                public void paint(final Graphics g)
1837                                {
1838                                        final int cw = this.getWidth();
1839                                        final int ch = this.getHeight();
1840                                        if (img.getWidth() < cw && img.getHeight() < ch)
1841                                        {
1842                                                final int x = (cw - img.getWidth()) / 2;
1843                                                final int y = (ch - img.getHeight()) / 2;
1844                                                g.drawImage(img, x, y, img.getWidth(),
1845                                                                img.getHeight(), f);
1846                                        }
1847                                        else if (img.getWidth() > img.getHeight())
1848                                        {
1849                                                final float sf = (float) cw / (float) img.getWidth();
1850                                                final int h = Math.round(sf * img.getHeight());
1851                                                g.drawImage(img, 0, (ch - h) / 2, cw, h, f);
1852                                        }
1853                                        else
1854                                        {
1855                                                final float sf = (float) ch / (float) img.getHeight();
1856                                                final int w = Math.round(sf * img.getWidth());
1857                                                g.drawImage(img, (cw - w) / 2, 0, w, ch, f);
1858                                        }
1859                                        // TODO: scale image proportionally and draw centered
1860
1861                                }
1862                        };
1863                        c.setSize(200, 200);
1864                        c.setPreferredSize(new Dimension(c.getWidth(), c.getHeight()));
1865                        container.add(c);
1866                }
1867                f.setSize(new Dimension(840, 600));
1868                f.setPreferredSize(new Dimension(840, 600));
1869
