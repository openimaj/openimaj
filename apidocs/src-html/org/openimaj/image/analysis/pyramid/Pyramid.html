030package org.openimaj.image.analysis.pyramid;
031
032import java.util.ArrayList;
033import java.util.Iterator;
034import java.util.List;
035
036import org.openimaj.image.FImage;
037import org.openimaj.image.Image;
038import org.openimaj.image.analyser.ImageAnalyser;
039import org.openimaj.image.processor.SinglebandImageProcessor;
040
041/**
042 * An image pyramid consisting of a stack of octaves.
043 * 
044 * Octaves are processed by an OctaveProcessor as they are created
045 * if the processor is set in the options object.
046 * 
047 * The pyramid will only hold onto its octaves if either the 
048 * keepOctaves option is set to true, or if a PyramidProcessor is
049 * set in the options. The PyramidProcessor will called after all
050 * the octaves are created.
051 * 
052 * Pyramids are Iterable for easy access to the octaves; however this
053 * will only work if the pyramid has already been populated with the
054 * octaves retained.
055 * 
056 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
057 *
058 * @param <OPTIONS> Type of options object
059 * @param <OCTAVE> Type of underlying octave
060 * @param <IMAGE> Type of underlying image
061 */
062public abstract class Pyramid<
063                OPTIONS extends PyramidOptions<OCTAVE,IMAGE>, 
064                OCTAVE extends Octave<OPTIONS,?,IMAGE>, 
065                IMAGE extends Image<?,IMAGE> & SinglebandImageProcessor.Processable<Float,FImage,IMAGE>> 
066        implements 
067                ImageAnalyser<IMAGE>, Iterable<OCTAVE> 
068{
069        /**
070         * Options for the pyramid
071         */
072        protected OPTIONS options;
073        
074        /**
075         * A list of all the octaves should you want to keep them.
076         * @see PyramidOptions#keepOctaves 
077         */
078        protected List<OCTAVE> octaves;
079        
080        /**
081         * Construct a Pyramid with the given options.
082         * @param options the options
083         */
084        public Pyramid(OPTIONS options) {
085                this.options = options;
086                
087                if (options.keepOctaves || options.pyramidProcessor!=null)
088                        octaves = new ArrayList<OCTAVE>();
089        }
090        
091        /**
092         * Process the image and construct a pyramid applying any specified
093         * OctaveProcessor and/or PyramidProcessor along the way. If a 
094         * PyramidProcessor is specified or the options have keepOctaves set 
095         * to true, then the octaves of the pyramid will be retained.
096         * 
097         * @param img image to build pyramid from.
098         */
099        public abstract void process(IMAGE img);
100        
101        /* (non-Javadoc)
102         * @see org.openimaj.image.analyser.ImageAnalyser#analyseImage(org.openimaj.image.Image)
103         */
104        @Override
105        public void analyseImage(IMAGE image) {
106                process(image);
107        }
108
109        /**
110         * Get the options used to initialise this pyramid
111         * @return the options
112         */
113        public OPTIONS getOptions() {
114                return options;
115        }
116        
117        /**
118         * Set the options used to initialise this pyramid
119         * @param options the options
120         */
121        public void setOptions(OPTIONS options) {
122                this.options = options;
123        }
124
125        /**
126         * Get the octaves of this pyramid if they have
127         * been retained by the processing.
128         * 
129         * @return the octaves
130         */
131        public List<OCTAVE> getOctaves() {
132                return octaves;
133        }
134
135        /* (non-Javadoc)
136         * @see java.lang.Iterable#iterator()
137         */
138        @Override
139        public Iterator<OCTAVE> iterator() {
140                if (octaves == null) return null;
141                return octaves.iterator();
142        }
143}
