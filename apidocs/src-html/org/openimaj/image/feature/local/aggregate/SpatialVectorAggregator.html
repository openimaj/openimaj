030package org.openimaj.image.feature.local.aggregate;
031
032import java.util.List;
033
034import org.openimaj.feature.FeatureVector;
035import org.openimaj.feature.local.LocalFeature;
036import org.openimaj.feature.local.Location;
037
038/**
039 * Interface describing an object that can convert a list of local features from
040 * a single image into an aggregated vector form, using both the featurevector
041 * and spatial location of each local feature. An example use is to create
042 * aggregate feature vectors that encode go beyond simple
043 * {@link VectorAggregator}s in that they additionally encode spatial
044 * information.
045 * 
046 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
047 * 
048 * @param <FEATURE>
049 *            The type of {@link FeatureVector} of the {@link LocalFeature}s
050 *            that can be processed.
051 * @param <LOCATION>
052 *            The type of {@link Location} of the {@link LocalFeature}s that can
053 *            be processed.
054 * @param <BOUNDS>
055 *            The spatial bounds in which the {@link LocalFeature}s were
056 *            extracted.
057 */
058public interface SpatialVectorAggregator<FEATURE extends FeatureVector, LOCATION extends Location, BOUNDS> {
059        /**
060         * Aggregate the given features into a vector. The features are assumed to
061         * have spatial locations within the given bounds; typically this might be
062         * the bounds rectangle of the image from which the features were extracted.
063         * 
064         * @param features
065         *            the features to aggregate
066         * @param bounds
067         *            the bounds in which the features were extracted
068         * @return the aggregated vector
069         */
070        public FeatureVector aggregate(List<? extends LocalFeature<? extends LOCATION, ? extends FEATURE>> features,
071                        BOUNDS bounds);
072}
