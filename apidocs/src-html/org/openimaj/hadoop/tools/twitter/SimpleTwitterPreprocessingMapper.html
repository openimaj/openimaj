030package org.openimaj.hadoop.tools.twitter;
031
032import java.io.ByteArrayOutputStream;
033import java.io.IOException;
034import java.io.OutputStreamWriter;
035import java.io.PrintWriter;
036import java.util.List;
037
038import org.apache.hadoop.io.LongWritable;
039import org.apache.hadoop.io.NullWritable;
040import org.apache.hadoop.io.Text;
041import org.apache.hadoop.mapreduce.Mapper;
042import org.kohsuke.args4j.CmdLineException;
043import org.openimaj.hadoop.tools.HadoopToolsUtil;
044import org.openimaj.tools.twitter.modes.preprocessing.TwitterPreprocessingMode;
045import org.openimaj.tools.twitter.options.AbstractTwitterPreprocessingToolOptions;
046import org.openimaj.twitter.USMFStatus;
047
048/**
049 * This mapper loads arguments for the
050 * {@link AbstractTwitterPreprocessingToolOptions} from the
051 * {@link HadoopTwitterPreprocessingTool#ARGS_KEY} variable (once per in memory
052 * mapper) and uses these to preprocess tweets.
053 *
054 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
055 *
056 */
057public class SimpleTwitterPreprocessingMapper extends Mapper<LongWritable, Text, NullWritable, Text> {
058        private HadoopTwitterPreprocessingToolOptions options = null;
059        private List<TwitterPreprocessingMode<?>> modes = null;
060
061        @Override
062        protected void setup(Mapper<LongWritable, Text, NullWritable, Text>.Context context) throws IOException,
063                        InterruptedException
064        {
065                try {
066                        final String[] args = HadoopToolsUtil.decodeArgs(context.getConfiguration().getStrings(
067                                        HadoopTwitterPreprocessingTool.ARGS_KEY));
068                        options = new HadoopTwitterPreprocessingToolOptions(args);
069                        options.prepare();
070                        modes = options.preprocessingMode();
071                } catch (final CmdLineException e) {
072                        throw new IOException(e);
073                } catch (final Exception e) {
074                        throw new IOException(e);
075                }
076        }
077
078        @Override
079        protected void map(LongWritable key, Text value, Mapper<LongWritable, Text, NullWritable, Text>.Context context)
080                        throws java.io.IOException, InterruptedException
081        {
082                final USMFStatus status = new USMFStatus(options.statusType.type());
083                status.fillFromString(value.toString());
084                if (status.isInvalid())
085                        return;
086
087                if (options.preProcessesSkip(status))
088                        return;
089                for (final TwitterPreprocessingMode<?> mode : modes) {
090                        try {
091                                TwitterPreprocessingMode.results(status, mode);
092                        } catch (final Exception e) {
093                                System.err.println("Failed mode: " + mode);
094                        }
095                }
096                if (options.postProcessesSkip(status))
097                        return;
098                final ByteArrayOutputStream baos = new ByteArrayOutputStream();
099                final OutputStreamWriter ow = new OutputStreamWriter(baos, "UTF-8");
100                final PrintWriter outTweetWriter = new PrintWriter(ow);
101                try {
102                        options.ouputMode().output(options.convertToOutputFormat(status), outTweetWriter);
103                        outTweetWriter.flush();
104                        context.write(NullWritable.get(), new Text(baos.toByteArray()));
105                } catch (final Exception e) {
106                        System.err.println("Failed to write tweet: " + status.text);
107                        System.err.println("With error: ");
108                        e.printStackTrace();
109                }
110        }
111}
