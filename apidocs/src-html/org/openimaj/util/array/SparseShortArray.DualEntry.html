455        public double dotProduct(SparseShortArray vector) {
456                double sum = 0;
457
458                int tused = this.used();
459                int vused = vector.used();
460
461                SparseShortArray smaller = (tused < vused ? this : vector);
462                SparseShortArray larger = (tused < vused ? vector : this);
463
464                for (Entry e : smaller.entries()) {
465                        sum += e.value * larger.get(e.index);
466                }
