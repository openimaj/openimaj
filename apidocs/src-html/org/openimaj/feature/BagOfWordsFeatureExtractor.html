030package org.openimaj.feature;
031
032import gnu.trove.impl.Constants;
033import gnu.trove.map.hash.TObjectIntHashMap;
034
035import java.util.List;
036
037import org.openimaj.util.array.SparseHashedFloatArray;
038
039/**
040 * An extractor which gives {@link SparseFloatFV} instances for a list of words.
041 * This is a simple unweighted count.
042 * 
043 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
044 */
045public class BagOfWordsFeatureExtractor implements FeatureExtractor<SparseFloatFV, List<String>> {
046        private List<String> dict;
047        private TObjectIntHashMap<String> lookup;
048
049        /**
050         * A set of words which are used for their index and therefore feature
051         * vector entry of a given word.
052         * 
053         * @param dictionary
054         */
055        public BagOfWordsFeatureExtractor(List<String> dictionary) {
056                this.dict = dictionary;
057                this.lookup = new TObjectIntHashMap<String>(Constants.DEFAULT_CAPACITY, Constants.DEFAULT_LOAD_FACTOR, -1);
058                int index = 0;
059                for (final String string : dictionary) {
060                        lookup.put(string, index++);
061                }
062        }
063
064        @Override
065        public SparseFloatFV extractFeature(List<String> object) {
066                final SparseHashedFloatArray values = new SparseHashedFloatArray(this.dict.size());
067                for (final String string : object) {
068                        final int index = asIndex(string);
069                        if (index < 0)
070                                continue;
071                        values.increment(index, 1f);
072                }
073                return new SparseFloatFV(values);
074        }
075
076        private int asIndex(String string) {
077                final int found = this.lookup.get(string);
078                return found;
079        }
080}
