030package org.openimaj.docs.tutorial.fund.images.clustseg;
031
032import java.io.IOException;
033import java.net.URL;
034import java.util.Arrays;
035import java.util.List;
036
037import org.openimaj.image.DisplayUtilities;
038import org.openimaj.image.ImageUtilities;
039import org.openimaj.image.MBFImage;
040import org.openimaj.image.colour.ColourSpace;
041import org.openimaj.image.connectedcomponent.GreyscaleConnectedComponentLabeler;
042import org.openimaj.image.pixel.ConnectedComponent;
043import org.openimaj.image.pixel.PixelSet;
044import org.openimaj.image.typography.hershey.HersheyFont;
045import org.openimaj.ml.clustering.FloatCentroidsResult;
046import org.openimaj.ml.clustering.assignment.HardAssigner;
047import org.openimaj.ml.clustering.kmeans.FloatKMeans;
048
049/**
050 * OpenIMAJ Hello world!
051 * 
052 */
053public class App {
054        /**
055         * Main method
056         * 
057         * @param args
058         * @throws IOException
059         */
060        public static void main(String[] args) throws IOException {
061                // Load the image
062                MBFImage input = ImageUtilities.readMBF(new URL("http://static.openimaj.org/media/tutorial/sinaface.jpg"));
063
064                input = ColourSpace.convert(input, ColourSpace.CIE_Lab);
065
066                final FloatKMeans cluster = FloatKMeans.createExact(3, 2);
067
068                final float[][] imageData = input.getPixelVectorNative(new float[input.getWidth() * input.getHeight()][3]);
069
070                final FloatCentroidsResult result = cluster.cluster(imageData);
071
072                final float[][] centroids = result.centroids;
073                for (final float[] fs : centroids) {
074                        System.out.println(Arrays.toString(fs));
075                }
076
077                final HardAssigner<float[], ?, ?> assigner = result.defaultHardAssigner();
078                for (int y = 0; y < input.getHeight(); y++) {
079                        for (int x = 0; x < input.getWidth(); x++) {
080                                final float[] pixel = input.getPixelNative(x, y);
081                                final int centroid = assigner.assign(pixel);
082                                input.setPixelNative(x, y, centroids[centroid]);
083                        }
084                }
085
086                input = ColourSpace.convert(input, ColourSpace.RGB);
087                DisplayUtilities.display(input);
088
089                final GreyscaleConnectedComponentLabeler labeler = new GreyscaleConnectedComponentLabeler();
090                final List<ConnectedComponent> components = labeler.findComponents(input.flatten());
091
092                int i = 0;
093                for (final PixelSet comp : components) {
094                        if (comp.calculateArea() < 50)
095                                continue;
096                        input.drawText("Point:" + (i++), comp.calculateCentroidPixel(), HersheyFont.TIMES_MEDIUM, 20);
097                }
098
099                DisplayUtilities.display(input);
100        }
101}
