033package org.openimaj.demos.core;
034
035import java.awt.Font;
036
037import org.openimaj.demos.Demo;
038import org.openimaj.image.DisplayUtilities;
039import org.openimaj.image.MBFImage;
040import org.openimaj.image.colour.RGBColour;
041import org.openimaj.image.typography.general.GeneralFont;
042
043/**
044 *      @author David Dupplaw (dpd@ecs.soton.ac.uk)
045 *      @created 5th November 2011
046 */
047@Demo(
048                author = "David Dupplaw",
049                description = "Demonstrates some the OpenIMAJ typography",
050                keywords = { "fonts" },
051                title = "Fonts"
052                )
053public class Fonts
054{
055        /**
056         *      Construct the fonts demo.
057         */
058        public Fonts()
059        {
060                final MBFImage img = new MBFImage( 800, 600, 3 );
061                img.drawText( "OOpenIMAJ", 20, 100,
062                                new GeneralFont("Arial", Font.PLAIN ), 120, RGBColour.WHITE );
063                img.drawText( "is Awesome!", 20, 220,
064                                new GeneralFont("Courier", Font.PLAIN ), 120, RGBColour.WHITE );
065                img.drawText( "HHope you agree", 20, 400,
066                                new GeneralFont("Comic Sans MS", Font.PLAIN ), 50, RGBColour.WHITE );
067
068                DisplayUtilities.display( img );
069        }
070
071        /**
072         *      Default main
073         *  @param args Command-line arguments
074         */
075        public static void main(final String[] args)
076        {
077                new Fonts();
078        }
079}
