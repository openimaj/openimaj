093            IntDoublePair choose_split(final double [][] pnts, final IntArrayView inds) {
094                int D = pnts[0].length;
095                
096                // Find mean & variance of each dimension.
097                double [] sum_x = new double[D];
098                double [] sum_xx = new double[D];
099                
100                int count = Math.min(inds.size(), varest_max_points);
101                for (int n=0; n<count; ++n) {
102                    for (int d=0; d<D; ++d) {
103                        sum_x[d]  += pnts[inds.getFast(n)][d];
104                        sum_xx[d] += (pnts[inds.getFast(n)][d]*pnts[inds.getFast(n)][d]);
105                    }
106                }
107
108                DoubleIntPair[] var_dim = new DoubleIntPair[D];
109                for (int d=0; d < D; ++d) {
110                        var_dim[d] = new DoubleIntPair();
111                    if (count <= 1)
112                        var_dim[d].first = 0;
113                    else
114                        var_dim[d].first = (sum_xx[d] - ((double)1/count)*sum_x[d]*sum_x[d])/(count - 1);
115                    var_dim[d].second = d;
116                }
117
118                // Partial sort makes a BIG difference to the build time.
119                int nrand = Math.min(varest_max_randsz, D);
120                Sorting.partial_sort(var_dim, 0, nrand, var_dim.length, new BinaryPredicate() {
121                                @Override
122                                public boolean apply(Object arg0, Object arg1) {
123                                        DoubleIntPair p1 = (DoubleIntPair) arg0;
124                                        DoubleIntPair p2 = (DoubleIntPair) arg1;
125                                        
126                                        if (p1.first > p2.first) return true;
127                                        if (p2.first > p1.first) return false;
128                                        return (p1.second > p2.second);
129                                }});
130                
131                int randd = var_dim[rng.nextIntFromTo(0, nrand-1)].second;
132                
133                return new IntDoublePair(randd, sum_x[randd]/count);
134            }

136            void split_points(final double [][] pnts, IntArrayView inds) {
137                IntDoublePair spl = choose_split(pnts, inds);
138
139                ((InternalNodeData)node_data).disc_dim = spl.first;
140                ((InternalNodeData)node_data).disc = spl.second;
141
142                int N = inds.size();
143                int l = 0;
144                int r = N;
145                while (l!=r) {
146                  if (pnts[inds.getFast(l)][((InternalNodeData)node_data).disc_dim] < ((InternalNodeData)node_data).disc) l++;
147                  else {
148                    r--;
149                    int t = inds.getFast(l);
150                    inds.setFast(l, inds.getFast(r));
151                    inds.setFast(r, t);
152                  }
153                }
154            
155                // If either partition is empty -> vectors identical!
156                if (l==0 || l==N) { l = N/2; } // The vectors are identical, so keep nlogn performance.
157
158                left = new DoubleKDTreeNode(pnts, inds.subView(0, l), rng);
159                
160                ((InternalNodeData)node_data).right = new DoubleKDTreeNode(pnts, inds.subView(l, N), rng);
161            }

186            void search(final double [] qu, PriorityQueue<DoubleObjectPair<DoubleKDTreeNode>> pri_branch, List<IntDoublePair> nns, boolean[] seen, double [][] pnts, double mindsq)
187            {
188                DoubleKDTreeNode cur = this;
189                DoubleKDTreeNode other = null;
190
191                while (!cur.is_leaf()) { // Follow best bin first until we hit a leaf
192                        double diff = qu[((InternalNodeData)cur.node_data).disc_dim] - ((InternalNodeData)cur.node_data).disc;
193
194                    if (diff < 0) {
195                        other = ((InternalNodeData)cur.node_data).right;
196                        cur = cur.left;
197                    }
198                    else {
199                        other = cur.left;
200                        cur = ((InternalNodeData)cur.node_data).right;
201                    }
202
203                    pri_branch.add(new DoubleObjectPair<DoubleKDTreeNode>(mindsq + diff*diff, other));
204                }
205
206                int [] cur_inds = ((LeafNodeData)cur.node_data).indices;
207                int ncur_inds = cur_inds.length;
208                
209                int i;
210                double [] dsq = new double[1];
211                for (i = 0; i < ncur_inds; ++i) {
212                        int ci = cur_inds[i];
213                    if (!seen[ci]) {
214                        DoubleNearestNeighbours.distanceFunc(qu, new double[][] {pnts[ci]}, dsq);
215                        
216                        nns.add(new IntDoublePair(ci, dsq[0]));
217                        
218                        seen[ci] = true;
219                    }
220                }
221            }
