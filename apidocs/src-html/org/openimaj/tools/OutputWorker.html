030package org.openimaj.tools;
031
032import java.io.File;
033import java.io.FileWriter;
034import java.io.IOException;
035import java.io.Writer;
036
037import org.kohsuke.args4j.CmdLineException;
038import org.openimaj.citation.ReferenceListener;
039import org.openimaj.citation.annotation.output.StandardFormatters;
040
041/**
042 * Helper class for outputting the bibliography information requested in the
043 * {@link ReferencesToolOpts}.
044 * 
045 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
046 * 
047 */
048public class OutputWorker implements Runnable {
049        ReferencesToolOpts options;
050
051        /**
052         * Construct with arguments string
053         * 
054         * @param args
055         *            the arguments
056         * @throws CmdLineException
057         */
058        public OutputWorker(String[] args) throws CmdLineException {
059                options = new ReferencesToolOpts(args);
060                options.validate();
061        }
062
063        private static void writeReferences(File file, StandardFormatters type) {
064                final String data =
065                                type.format(ReferenceListener.getReferences());
066
067                Writer writer = null;
068                try {
069                        writer = new FileWriter(file);
070                        writer.append(data);
071                } catch (final IOException e) {
072                        System.err.println("Error writing references file: " + file);
073                        e.printStackTrace();
074                } finally {
075                        if (writer != null)
076                                try {
077                                        writer.close();
078                                } catch (final IOException e) {
079                                }
080                }
081        }
082
083        private static void printReferences(StandardFormatters type) {
084                final String data =
085                                type.format(ReferenceListener.getReferences());
086
087                System.out.println(data);
088        }
089
090        @Override
091        public void run() {
092                if (options.printBibtex)
093                        printReferences(StandardFormatters.BIBTEX);
094                if (options.printHTML)
095                        printReferences(StandardFormatters.HTML);
096                if (options.printText)
097                        printReferences(StandardFormatters.STRING);
098
099                if (options.bibtexFile != null)
100                        writeReferences(options.bibtexFile, StandardFormatters.BIBTEX);
101                if (options.htmlFile != null)
102                        writeReferences(options.htmlFile, StandardFormatters.HTML);
103                if (options.textFile != null)
104                        writeReferences(options.textFile, StandardFormatters.STRING);
105        }
106}
