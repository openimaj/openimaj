030package org.openimaj.stream.functions.twitter;
031
032import java.net.MalformedURLException;
033import java.net.URL;
034import java.util.ArrayList;
035import java.util.HashSet;
036import java.util.List;
037import java.util.Set;
038import java.util.regex.Matcher;
039import java.util.regex.Pattern;
040
041import org.apache.log4j.Logger;
042import org.openimaj.text.nlp.patterns.URLPatternProvider;
043import org.openimaj.util.function.MultiFunction;
044
045import twitter4j.MediaEntity;
046import twitter4j.Status;
047import twitter4j.URLEntity;
048
049/**
050 * This class implements a function that processes Twitter {@link Status}
051 * objects to extract all the mentioned URLs. URLs are extracted from both the
052 * entities field and the Tweet message body (by applying a regular expression).
053 *
054 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
055 */
056public class TwitterURLExtractor implements MultiFunction<Status, URL> {
057        private final static Logger logger = Logger.getLogger(TwitterURLExtractor.class);
058        private final static Pattern urlPattern = new URLPatternProvider().pattern();
059
060        @Override
061        public List<URL> apply(Status status) {
062                final Set<String> urls = new HashSet<String>();
063
064                // Add all the entries from entities.urls
065                if (status.getURLEntities() != null) {
066                        for (final URLEntity map : status.getURLEntities()) {
067                                String u = map.getExpandedURL();
068
069                                if (u == null) {
070                                        u = map.getURL();
071
072                                        if (u == null)
073                                                continue;
074                                }
075
076                                urls.add(u);
077                        }
078                }
079
080                // Add all the entries from media.urls
081                for (final MediaEntity map : status.getMediaEntities()) {
082                        String u = map.getMediaURL();
083
084                        if (u == null) {
085                                u = map.getMediaURLHttps();
086
087                                if (u == null)
088                                        u = map.getDisplayURL();
089
090                                if (u == null)
091                                        continue;
092                        }
093                        urls.add(u);
094                }
095
096                // Find the URLs in the raw text
097                final String text = status.getText();
098                if (text != null) { // why was text null?
099                        final Matcher matcher = urlPattern.matcher(text);
100
101                        while (matcher.find()) {
102                                String urlString = text.substring(matcher.start(), matcher.end());
103
104                                if (!urlString.contains("://"))
105                                        urlString = "http://" + urlString;
106
107                                urls.add(urlString);
108                        }
109                }
110
111                // get the final URLs
112                final ArrayList<URL> finalUrls = new ArrayList<URL>();
113                for (final String u : urls) {
114                        try {
115                                finalUrls.add(new URL(u));
116                        } catch (final MalformedURLException e) {
117                                logger.warn("ignoring URL ", e);
118                        }
119                }
120                return finalUrls;
121        }
122}
