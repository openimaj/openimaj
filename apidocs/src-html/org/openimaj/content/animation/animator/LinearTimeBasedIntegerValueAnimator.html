036/**
037 * A {@link TimeBasedValueAnimator} that linearly animates a int value between two 
038 * values over a given time.
039 * 
040 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
041 * @author David Dupplaw (dpd@ecs.soton.ac.uk)
042 */
043public class LinearTimeBasedIntegerValueAnimator extends TimeBasedValueAnimator<Integer> 
044{
045        /**
046         * Construct a {@link LinearTimeBasedIntegerValueAnimator} with the given
047         * start and finish values, and the given duration in millseconds. 
048         * The animation starts on the first call to makeNextValue()
049         * and completes once duration ticks have been reached. 
050         * 
051         * @param start start value
052         * @param finish stop value
053         * @param duration in milliseconds
054         */
055        public LinearTimeBasedIntegerValueAnimator( int start, int finish, long duration) {
056                super( start, finish, duration );
057        }
058
059        /**
060         *      Calculates the value of the int based on the start, end and current
061         *      percentage.
062         *      @param pc The percentage
063         *      @return A int that is <code>pc</code> along between start and finish
064         */
065        @Override
066        public Integer calculateValue( double pc )
067        {
068                return (int)((endValue-startValue)*pc + startValue);
069        }
