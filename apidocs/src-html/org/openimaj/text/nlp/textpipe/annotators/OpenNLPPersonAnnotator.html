030package org.openimaj.text.nlp.textpipe.annotators;
031
032import java.io.IOException;
033import java.io.InputStream;
034import java.util.List;
035
036import opennlp.tools.namefind.NameFinderME;
037import opennlp.tools.namefind.TokenNameFinderModel;
038import opennlp.tools.util.Span;
039
040import org.openimaj.text.nlp.textpipe.annotations.AnnotationUtils;
041import org.openimaj.text.nlp.textpipe.annotations.RawTextAnnotation;
042import org.openimaj.text.nlp.textpipe.annotations.SentenceAnnotation;
043import org.openimaj.text.nlp.textpipe.annotations.TokenAnnotation;
044
045/**
046 * Uses a {@link TokenNameFinderModel} instnace to instanciate a {@link NameFinderME}
047 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
048 *
049 */
050public class OpenNLPPersonAnnotator extends AbstractNEAnnotator {
051
052        /**
053         * Property name containing the person model
054         */
055        public static final String PERSON_MODEL_PROP = "org.openimaj.text.opennlp.models.person";
056        NameFinderME nameFinder;
057
058        /**
059         * loads a model from {@link OpenNLPPersonAnnotator#PERSON_MODEL_PROP}
060         */
061        public OpenNLPPersonAnnotator() {
062                super();
063                TokenNameFinderModel model = null;
064                String resourceLocation = System.getProperty(PERSON_MODEL_PROP);
065                InputStream modelIn = OpenNLPPersonAnnotator.class.getResourceAsStream(resourceLocation);
066
067                try {
068                        model = new TokenNameFinderModel(modelIn);
069                } catch (IOException e) {
070                        e.printStackTrace();
071                } finally {
072                        if (modelIn != null) {
073                                try {
074                                        modelIn.close();
075                                } catch (IOException e) {
076                                }
077                        }
078                }
079                nameFinder = new NameFinderME(model);
080        }
081
082        @Override
083        void performAnnotation(RawTextAnnotation annotation)
084                        throws MissingRequiredAnnotationException {
085
086
087                          for (SentenceAnnotation sentence : annotation.getAnnotationsFor(SentenceAnnotation.class)) {
088                                  List<TokenAnnotation> atoks = sentence.getAnnotationsFor(TokenAnnotation.class);
089                                  List<String> toks = AnnotationUtils.getStringTokensFromTokenAnnotationList(atoks);
090                            Span nameSpans[] = nameFinder.find(AnnotationUtils.ListToArray(toks));
091                            for(Span s :nameSpans){
092//                              NamedEntityAnnotation nea = new NamedEntityAnnotation();
093//                              NamedEntity ne = new NamedEntity();
094                                for(int i = s.getStart();i<s.getEnd();i++){
095                                        atoks.get(i).addAnnotation(annotation);
096                                }
097                            }
098                          }
099                          nameFinder.clearAdaptiveData();
100
101        }
102
103        @Override
104        void checkForRequiredAnnotations(RawTextAnnotation annotation)
105                        throws MissingRequiredAnnotationException {
106                // TODO Auto-generated method stub
107
108        }
109
110}
