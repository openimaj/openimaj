151        public static int [] getPath(int index, int depth, int K) {
152                int [] path = new int[depth];
153
154                for (int i=0; i<depth; i++) {
155                        int v = ipow(K, depth-1-i);
156
157                        for (int j=0; j<K; j++) {       
158                                int vp = v * j;
159                                int vpn = v * (j+1);
160
161                                if (index < vpn) {
162                                        path[i] = j;
163                                        index -= vp;
164                                        break;
165                                }
166                        }
167                }
168
169                return path;
170        }
