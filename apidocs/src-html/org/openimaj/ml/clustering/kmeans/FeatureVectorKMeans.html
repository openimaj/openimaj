057/**
058 * Fast, parallel implementation of the K-Means algorithm with support for
059 * bigger-than-memory data. Various flavors of K-Means are supported through the
060 * selection of different subclasses of {@link ObjectNearestNeighbours}; for
061 * example, exact K-Means can be achieved using an
062 * {@link ObjectNearestNeighboursExact}. The specific choice of
063 * nearest-neighbour object is controlled through the
064 * {@link NearestNeighboursFactory} provided to the {@link KMeansConfiguration}
065 * used to construct instances of this class. The choice of
066 * {@link ObjectNearestNeighbours} affects the speed of clustering; using
067 * approximate nearest-neighbours algorithms for the K-Means can produces
068 * comparable results to the exact KMeans algorithm in much shorter time. The
069 * choice and configuration of {@link ObjectNearestNeighbours} can also control
070 * the type of distance function being used in the clustering.
071 * <p>
072 * The algorithm is implemented as follows: Clustering is initiated using a
073 * {@link ByteKMeansInit} and is iterative. In each round, batches of samples
074 * are assigned to centroids in parallel. The centroid assignment is performed
075 * using the pre-configured {@link ObjectNearestNeighbours} instances created
076 * from the {@link KMeansConfiguration}. Once all samples are assigned new
077 * centroids are calculated and the next round started. Data point pushing is
078 * performed using the same techniques as center point assignment.
079 * <p>
080 * This implementation is able to deal with larger-than-memory datasets by
081 * streaming the samples from disk using an appropriate {@link DataSource}. The
082 * only requirement is that there is enough memory to hold all the centroids
083 * plus working memory for the batches of samples being assigned.
084 *
085 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
086 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
087 *
088 * @param <T>
089 *            Type of object being clustered
090 */
091public class FeatureVectorKMeans<T extends FeatureVector> implements SpatialClusterer<FeatureVectorCentroidsResult<T>, T>
092{
093        private static class CentroidAssignmentJob<T extends FeatureVector> implements Callable<Boolean> {
094                private final DataSource<T> ds;
095                private final int startRow;
096                private final int stopRow;
097                private final ObjectNearestNeighbours<T> nno;
098                private final double[][] centroids_accum;
099                private final int[] counts;
100
101                public CentroidAssignmentJob(DataSource<T> ds, int startRow, int stopRow, ObjectNearestNeighbours<T> nno,
102                                double[][] centroids_accum, int[] counts)
103                {
104                        this.ds = ds;
105                        this.startRow = startRow;
106                        this.stopRow = stopRow;
107                        this.nno = nno;
108                        this.centroids_accum = centroids_accum;
109                        this.counts = counts;
110                }
111
112                @Override
113                public Boolean call() {
114                        try {
115                                final int D = ds.getData(0).length();
116
117                                final T[] points = ds.createTemporaryArray(stopRow - startRow);
118                                ds.getData(startRow, stopRow, points);
119
120                                final int[] argmins = new int[points.length];
121                                final float[] mins = new float[points.length];
122
123                                nno.searchNN(points, argmins, mins);
124
125                                synchronized (centroids_accum) {
126                                        for (int i = 0; i < points.length; ++i) {
127                                                final int k = argmins[i];
128                                                final double[] vector = points[i].asDoubleVector();
129                                                for (int d = 0; d < D; ++d) {
130                                                        centroids_accum[k][d] += vector[d];
131                                                }
132                                                counts[k] += 1;
133                                        }
134                                }
135                        } catch (final Exception e) {
136                                e.printStackTrace();
137                        }
138                        return true;
139                }
140        }
141

157        }

229        @Override
230        public FeatureVectorCentroidsResult<T> cluster(T[] data) {
231                final ArrayBackedDataSource<T> ds = new ArrayBackedDataSource<T>(data, rng) {
232                        @Override
233                        public int numDimensions() {
234                                return data[0].length();
235                        }
236                };
237
238                try {
239                        final Result<T> result = cluster(ds, conf.K);
240                        result.nn = conf.factory.create(result.centroids);
241
242                        return result;
243                } catch (final Exception e) {
244                        throw new RuntimeException(e);
245                }
246        }

250                final FeatureVectorCentroidsResult<T> clusters = this.cluster(data);
251                return new IndexClusters(clusters.defaultHardAssigner().assign(data)).clusters();

271        /**
272         * Initiate clustering with the given data and number of clusters.
273         * Internally this method constructs the array to hold the centroids and
274         * calls {@link #cluster(DataSource, Object)}.
275         *
276         * @param data
277         *            data source to cluster with
278         * @param K
279         *            number of clusters to find
280         * @return cluster centroids
281         */
282        protected Result<T> cluster(DataSource<T> data, int K) throws Exception {
283                final Result<T> result = new Result<T>();
284                result.centroids = data.createTemporaryArray(K);
285
286                init.initKMeans(data, result.centroids);
287
288                cluster(data, result);
289
290                return result;
291        }

293        /**
294         * Main clustering algorithm. A number of threads as specified are started
295         * each containing an assignment job and a reference to the same set of
296         * ObjectNearestNeighbours object (i.e. Exact or KDTree). Each thread is
297         * added to a job pool and started in parallel. A single accumulator is
298         * shared between all threads and locked on update.
299         *
300         * @param data
301         *            the data to be clustered
302         * @param centroids
303         *            the centroids to be found
304         */
305        protected void cluster(DataSource<T> data, Result<T> result) throws Exception {
306                final T[] centroids = result.centroids;
307                final int K = centroids.length;
308                final int D = centroids[0].length();
309                final int N = data.size();
310                final double[][] centroids_accum = new double[K][D];
311                final int[] new_counts = new int[K];
312
313                final ExecutorService service = conf.threadpool;
314
315                for (int i = 0; i < conf.niters; i++) {
316                        System.err.println("Iteration " + i);
317                        for (int j = 0; j < K; j++)
318                                Arrays.fill(centroids_accum[j], 0);
319                        Arrays.fill(new_counts, 0);
320
321                        final ObjectNearestNeighbours<T> nno = conf.factory.create(centroids);
322
323                        final List<CentroidAssignmentJob<T>> jobs = new ArrayList<CentroidAssignmentJob<T>>();
324                        for (int bl = 0; bl < N; bl += conf.blockSize) {
325                                final int br = Math.min(bl + conf.blockSize, N);
326                                jobs.add(new CentroidAssignmentJob<T>(data, bl, br, nno, centroids_accum, new_counts));
327                        }
328
329                        service.invokeAll(jobs);
330
331                        for (int k = 0; k < K; ++k) {
332                                if (new_counts[k] == 0) {
333                                        // If there's an empty cluster we replace it with a random
334                                        // point.
335                                        new_counts[k] = 1;
336
337                                        final T[] rnd = data.createTemporaryArray(1);
338                                        data.getRandomRows(rnd);
339
340                                        final Cloner cloner = new Cloner();
341                                        centroids[k] = cloner.deepClone(rnd[0]);
342                                } else {
343                                        for (int d = 0; d < D; ++d) {
344                                                centroids[k].setFromDouble(d, centroids_accum[k][d] / new_counts[k]);
345                                        }
346                                }
347                        }
348                }
349        }

369                try {
370                        final Result<T> result = cluster(ds, conf.K);
371                        result.nn = conf.factory.create(result.centroids);

375                        throw new RuntimeException(e);

398        /**
399         * Convenience method to quickly create an exact {@link ByteKMeans}. All
400         * parameters other than the number of clusters are set at their defaults,
401         * but can be manipulated through the configuration returned by
402         * {@link #getConfiguration()}.
403         * <p>
404         * Euclidean distance is used to measure the distance between points.
405         *
406         * @param K
407         *            the number of clusters
408         * @param distance
409         *            the distance measure
410         * @return a {@link ByteKMeans} instance configured for exact k-means
411         */
412        public static <T extends FeatureVector> FeatureVectorKMeans<T> createExact(int K,
413                        DistanceComparator<? super T> distance)
414        {
415                final KMeansConfiguration<ObjectNearestNeighbours<T>, T> conf =
416                                new KMeansConfiguration<ObjectNearestNeighbours<T>, T>(K, new ObjectNearestNeighboursExact.Factory<T>(
417                                                distance));
418
419                return new FeatureVectorKMeans<T>(conf);
420        }

422        /**
423         * Convenience method to quickly create an exact {@link ByteKMeans}. All
424         * parameters other than the number of clusters and number of iterations are
425         * set at their defaults, but can be manipulated through the configuration
426         * returned by {@link #getConfiguration()}.
427         * <p>
428         * Euclidean distance is used to measure the distance between points.
429         *
430         * @param K
431         *            the number of clusters
432         * @param distance
433         *            the distance measure
434         * @param niters
435         *            maximum number of iterations
436         * @return a {@link ByteKMeans} instance configured for exact k-means
437         */
438        public static <T extends FeatureVector> FeatureVectorKMeans<T> createExact(int K,
439                        DistanceComparator<? super T> distance, int niters)
440        {
441                final KMeansConfiguration<ObjectNearestNeighbours<T>, T> conf =
442                                new KMeansConfiguration<ObjectNearestNeighbours<T>, T>(K,
443                                                new ObjectNearestNeighboursExact.Factory<T>(distance),
444                                                niters);
445
446                return new FeatureVectorKMeans<T>(conf);
447        }
