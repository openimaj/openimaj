030package org.openimaj.citation.annotation.input;
031
032import java.io.IOException;
033import java.io.InputStreamReader;
034import java.io.Reader;
035
036import org.jbibtex.BibTeXDatabase;
037import org.jbibtex.BibTeXEntry;
038import org.jbibtex.BibTeXParser;
039import org.jbibtex.ParseException;
040import org.openimaj.citation.annotation.Reference;
041import org.openimaj.citation.annotation.mock.MockReference;
042import org.openimaj.citation.annotation.output.StandardFormatters;
043
044/**
045 * Helper tool to convert bibtex to {@link Reference} annotations.
046 * 
047 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
048 * 
049 */
050public class BibtexToReference {
051        /**
052         * Helper tool to convert bibtex to {@link Reference} annotations. Reads
053         * from stdin, writes to stdout.
054         * 
055         * @param args
056         *            not used
057         * @throws IOException
058         * @throws ParseException
059         */
060        public static void main(String[] args) throws IOException, ParseException {
061                System.out.println("Enter bibtex record(s), followed by ctrl-d: ");
062
063                final Reader reader = new InputStreamReader(System.in);
064
065                final BibTeXParser parser = new BibTeXParser();
066                final BibTeXDatabase database = parser.parse(reader);
067
068                System.out.println();
069
070                for (final BibTeXEntry entry : database.getEntries().values()) {
071                        final Reference r = MockReference.makeReference(entry);
072                        System.out.println(StandardFormatters.REFERENCE_ANNOTATION.format(r));
073                }
074        }
075}
