17  package org.openimaj.web.scraping;
18  
19  import java.net.URL;
20  import java.util.List;
21  
22  /**
23   * Site specific consumers answer whether they can handle a URL and when asked
24   * handle the URL, returning another URL from which the data can be downloaded.
25   * This interface doesn't specify exactly what the data is, but specific
26   * implementations will usually only apply to a single data modality.
27   * <p>
28   * Typical uses are for the downloading of images on a URL to an image hosting
29   * site.
30   * 
31   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
32   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
33   * 
34   */
35  public interface SiteSpecificConsumer {
36  	/**
37  	 * Determine whether the given URL can be handled by this consumer.
38  	 * 
39  	 * @param url
40  	 *            the url to test
41  	 * @return true if the URL can be handled by this consumer; false otherwise.
42  	 */
43  	public boolean canConsume(URL url);
44  
45  	/**
46  	 * Get the data urls at the given URL.
47  	 * 
48  	 * @param url
49  	 *            the url to test
50  	 * @return A list of URLs to data items at the given URL
51  	 */
52  	public List<URL> consume(URL url);
53  }
