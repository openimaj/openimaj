17  package org.openimaj.web.scraping;
18  
19  import java.net.URL;
20  
21  /**
22   * Simple scraper that just uses the given css selector to find all
23   * relevant data in the page
24   * 
25   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
26   * 
27   */
28  public class SimpleHTMLScrapingConsumer extends HTMLScrapingSiteSpecificConsumer {
29  	private String linkContains;
30  	private String select;
31  
32  	/**
33  	 * @param linkContains
34  	 *            the link should contain this
35  	 * @param select
36  	 *            the css selector for the img
37  	 */
38  	public SimpleHTMLScrapingConsumer(String linkContains, String select) {
39  		this.linkContains = linkContains;
40  		this.select = select;
41  	}
42  
43  	@Override
44  	public boolean canConsume(URL url) {
45  		return url.getHost().contains(linkContains);
46  	}
47  
48  	@Override
49  	public String cssSelect() {
50  		return select;
51  	}
52  
53  }
